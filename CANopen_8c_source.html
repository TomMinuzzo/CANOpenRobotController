157  uint8_t CO_sendNMTcommand(CO_t *CO, uint8_t command, uint8_t nodeID){
158  if(NMTM_txBuff == 0){
159  /* error, CO_CANtxBufferInit() was not called for this buffer. */
160  return CO_ERROR_TX_UNCONFIGURED; /* -11 */
161  }
162  NMTM_txBuff->data[0] = command;
163  NMTM_txBuff->data[1] = nodeID;
164 
165  /* Apply NMT command also to this node, if set so. */
166  if(nodeID == 0 || nodeID == CO->NMT->nodeId){
167  switch(command){
168  case CO_NMT_ENTER_OPERATIONAL:
169  if((*CO->NMT->emPr->errorRegister) == 0) {
170  CO->NMT->operatingState = CO_NMT_OPERATIONAL;
171  }
172  break;
173  case CO_NMT_ENTER_STOPPED:
174  CO->NMT->operatingState = CO_NMT_STOPPED;
175  break;
176  case CO_NMT_ENTER_PRE_OPERATIONAL:
177  CO->NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
178  break;
179  case CO_NMT_RESET_NODE:
180  CO->NMT->resetCommand = CO_RESET_APP;
181  break;
182  case CO_NMT_RESET_COMMUNICATION:
183  CO->NMT->resetCommand = CO_RESET_COMM;
184  break;
185  }
186  }
187 
188  return CO_CANsend(CO->CANmodule[0], NMTM_txBuff); /* 0 = success */
189  }
190 #endif
