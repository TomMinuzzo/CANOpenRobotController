54 void taskMain_init(int fdEpoll, uint16_t *maxTime) {
55  struct epoll_event ev;
56  int flags;
57 
58  /* Prepare pipe for triggering events. For example, if new SDO request
59  * arrives from CAN network, CANrx callback writes a byte into the pipe.
60  * This immediately triggers (via epoll) processing of SDO server, which
61  * generates response. Read and write ends of pipe are nonblocking.
62  * (See 'self pipe trick'.) */
63  if(pipe(taskMain.fdPipe) == -1)
64  CO_errExit("taskMain_init - pipe failed");
65 
66  flags = fcntl(taskMain.fdPipe[0], F_GETFL);
67  if(flags == -1)
68  CO_errExit("taskMain_init - fcntl-F_GETFL[0] failed");
69  flags |= O_NONBLOCK;
70  if(fcntl(taskMain.fdPipe[0], F_SETFL, flags) == -1)
71  CO_errExit("taskMain_init - fcntl-F_SETFL[0] failed");
72 
73  flags = fcntl(taskMain.fdPipe[1], F_GETFL);
74  if(flags == -1)
75  CO_errExit("taskMain_init - fcntl-F_GETFL[1] failed");
76  flags |= O_NONBLOCK;
77  if(fcntl(taskMain.fdPipe[1], F_SETFL, flags) == -1)
78  CO_errExit("taskMain_init - fcntl-F_SETFL[1] failed");
79 
80  /* get file descriptor for timer */
81  taskMain.fdTmr = timerfd_create(CLOCK_MONOTONIC, 0);
82  if(taskMain.fdTmr == -1)
83  CO_errExit("taskMain_init - timerfd_create failed");
84 
85  /* add events for epoll */
86  ev.events = EPOLLIN;
87  ev.data.fd = taskMain.fdPipe[0];
88  if(epoll_ctl(fdEpoll, EPOLL_CTL_ADD, taskMain.fdPipe[0], &ev) == -1)
89  CO_errExit("taskMain_init - epoll_ctl CANrx failed");
90 
91  ev.events = EPOLLIN;
92  ev.data.fd = taskMain.fdTmr;
93  if(epoll_ctl(fdEpoll, EPOLL_CTL_ADD, taskMain.fdTmr, &ev) == -1)
94  CO_errExit("taskMain_init - epoll_ctl taskTmr failed");
95 
96  /* Prepare timer, use no interval, delay time will be set each cycle. */
97  taskMain.tmrSpec.it_interval.tv_sec = 0;
98  taskMain.tmrSpec.it_interval.tv_nsec = 0;
99 
100  taskMain.tmrSpec.it_value.tv_sec = 0;
101  taskMain.tmrSpec.it_value.tv_nsec = 1;
102 
103  if(timerfd_settime(taskMain.fdTmr, 0, &taskMain.tmrSpec, NULL) != 0)
104  CO_errExit("taskMain_init - timerfd_settime failed");
105 
106  taskMain.tmr1msPrev = 0;
107  taskMain.maxTime = maxTime;
108 }

118 bool_t taskMain_process(int fd, CO_NMT_reset_cmd_t *reset, uint16_t timer1ms) {
119  bool_t wasProcessed = true;
120 
121  /* Signal from pipe, consume all bytes. */
122  if(fd == taskMain.fdPipe[0]) {
123  for(;;) {
124  char ch;
125  if(read(taskMain.fdPipe[0], &ch, 1) == -1) {
126  if (errno == EAGAIN)
127  break; /* No more bytes. */
128  else
129  CO_error(0x21100000L + errno);
130  }
131  }
132  }
133 
134  /* Timer expired. */
135  else if(fd == taskMain.fdTmr) {
136  uint64_t tmrExp;
137  if(read(taskMain.fdTmr, &tmrExp, sizeof(tmrExp)) != sizeof(uint64_t))
138  CO_error(0x21200000L + errno);
139  }
140  else {
141  wasProcessed = false;
142  }
143 
144  /* Process mainline. */
145  if(wasProcessed) {
146  uint16_t timer1msDiff;
147  uint16_t timerNext = 50;
148 
149  /* Calculate time difference */
150  timer1msDiff = timer1ms - taskMain.tmr1msPrev;
151  taskMain.tmr1msPrev = timer1ms;
152 
153  /* Calculate maximum interval in milliseconds (informative) */
154  if(taskMain.maxTime != NULL) {
155  if(timer1msDiff > *taskMain.maxTime) {
156  *taskMain.maxTime = timer1msDiff;
157  }
158  }
159 
160 
161  /* CANopen process */
162  *reset = CO_process(CO, timer1msDiff, &timerNext);
163 
164 
165  /* Set delay for next sleep. */
166  taskMain.tmrSpec.it_value.tv_nsec = (long)(++timerNext) * NSEC_PER_MSEC;
167  if(timerfd_settime(taskMain.fdTmr, 0, &taskMain.tmrSpec, NULL) == -1)
168  CO_error(0x21500000L + errno);
169 
170  }
171 
172  return wasProcessed;
173 }

194 void CANrx_taskTmr_init(int fdEpoll, long intervalns, uint16_t *maxTime) {
195  struct epoll_event ev;
196 
197  /* get file descriptors */
198  taskRT.fdRx0 = CO->CANmodule[0]->fd;
199 
200  taskRT.fdTmr = timerfd_create(CLOCK_MONOTONIC, 0);
201  if(taskRT.fdTmr == -1)
202  CO_errExit("CANrx_taskTmr_init - timerfd_create failed");
203 
204  /* add events for epoll */
205  ev.events = EPOLLIN;
206  ev.data.fd = taskRT.fdRx0;
207  if(epoll_ctl(fdEpoll, EPOLL_CTL_ADD, taskRT.fdRx0, &ev) == -1)
208  CO_errExit("CANrx_taskTmr_init - epoll_ctl CANrx failed");
209 
210  ev.events = EPOLLIN;
211  ev.data.fd = taskRT.fdTmr;
212  if(epoll_ctl(fdEpoll, EPOLL_CTL_ADD, taskRT.fdTmr, &ev) == -1)
213  CO_errExit("CANrx_taskTmr_init - epoll_ctl taskTmr failed");
214 
215  /* Prepare timer (one shot, each time calculate new expiration time) It is
216  * necessary not to use taskRT.tmrSpec.it_interval, because it is sliding. */
217  taskRT.tmrSpec.it_interval.tv_sec = 0;
218  taskRT.tmrSpec.it_interval.tv_nsec = 0;
219 
220  taskRT.tmrVal = &taskRT.tmrSpec.it_value;
221  if(clock_gettime(CLOCK_MONOTONIC, taskRT.tmrVal) != 0)
222  CO_errExit("CANrx_taskTmr_init - clock_gettime failed");
223 
224  if(timerfd_settime(taskRT.fdTmr, TFD_TIMER_ABSTIME, &taskRT.tmrSpec, NULL) != 0)
225  CO_errExit("CANrx_taskTmr_init - timerfd_settime failed");
226 
227  taskRT.intervalns = intervalns;
228  taskRT.intervalus = intervalns / 1000;
229  taskRT.maxTime = maxTime;
230 }

238 bool_t CANrx_taskTmr_process(int fd) {
239  bool_t wasProcessed = true;
240 
241  /* Get received CAN message. */
242  if(fd == taskRT.fdRx0) {
243  CO_CANrxWait(CO->CANmodule[0]);
244  }
245 
246  /* Execute taskTmr */
247  else if(fd == taskRT.fdTmr) {
248  uint64_t tmrExp;
249 
250  /* Wait for timer to expire */
251  if(read(taskRT.fdTmr, &tmrExp, sizeof(tmrExp)) != sizeof(uint64_t))
252  CO_error(0x22100000L + errno);
253 
254  /* Calculate maximum interval in microseconds (informative) */
255  if(taskRT.maxTime != NULL) {
256  struct timespec tmrMeasure;
257  if(clock_gettime(CLOCK_MONOTONIC, &tmrMeasure) == -1)
258  CO_error(0x22200000L + errno);
259  if(tmrMeasure.tv_sec == taskRT.tmrVal->tv_sec) {
260  long dt = tmrMeasure.tv_nsec - taskRT.tmrVal->tv_nsec;
261  dt /= 1000;
262  dt += taskRT.intervalus;
263  if(dt > 0xFFFF) {
264  *taskRT.maxTime = 0xFFFF;
265  }else if(dt > *taskRT.maxTime) {
266  *taskRT.maxTime = (uint16_t) dt;
267  }
268  }
269  }
270 
271  /* Calculate next shot for the timer */
272  taskRT.tmrVal->tv_nsec += taskRT.intervalns;
273  if(taskRT.tmrVal->tv_nsec >= NSEC_PER_SEC) {
274  taskRT.tmrVal->tv_nsec -= NSEC_PER_SEC;
275  taskRT.tmrVal->tv_sec++;
276  }
277  if(timerfd_settime(taskRT.fdTmr, TFD_TIMER_ABSTIME, &taskRT.tmrSpec, NULL) == -1)
278  CO_error(0x22300000L + errno);
279 
280 
281  /* Lock PDOs and OD */
282  CO_LOCK_OD();
283 
284  if(CO->CANmodule[0]->CANnormal) {
285  bool_t syncWas;
286 
287  /* Process Sync and read inputs */
288  syncWas = CO_process_SYNC_RPDO(CO, taskRT.intervalus);
289 
290  /* Further I/O or nonblocking application code may go here. */
291 
292  /* Write outputs */
293  CO_process_TPDO(CO, syncWas, taskRT.intervalus);
294  }
295 
296  /* Unlock */
297  CO_UNLOCK_OD();
298  }
299 
300  else {
301  wasProcessed = false;
302  }
303 
304  return wasProcessed;
305 }
