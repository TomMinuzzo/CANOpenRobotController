976 template <>
977 struct scalar_logistic_op<float> {
978  EIGEN_EMPTY_STRUCT_CTOR(scalar_logistic_op)
979  EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE float operator()(const float& x) const {
980  return packetOp(x);
981  }
982 
983  template <typename Packet> EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE
984  Packet packetOp(const Packet& _x) const {
985  const Packet cutoff_lower = pset1<Packet>(-9.f);
986  const Packet lt_mask = pcmp_lt<Packet>(_x, cutoff_lower);
987  const bool any_small = predux_any(lt_mask);
988 
989  // The upper cut-off is the smallest x for which the rational approximation evaluates to 1.
990  // Choosing this value saves us a few instructions clamping the results at the end.
991 #ifdef EIGEN_VECTORIZE_FMA
992  const Packet cutoff_upper = pset1<Packet>(15.7243833541870117f);
993 #else
994  const Packet cutoff_upper = pset1<Packet>(15.6437711715698242f);
995 #endif
996  const Packet x = pmin(_x, cutoff_upper);
997 
998  // The monomial coefficients of the numerator polynomial (odd).
999  const Packet alpha_1 = pset1<Packet>(2.48287947061529e-01f);
1000  const Packet alpha_3 = pset1<Packet>(8.51377133304701e-03f);
1001  const Packet alpha_5 = pset1<Packet>(6.08574864600143e-05f);
1002  const Packet alpha_7 = pset1<Packet>(1.15627324459942e-07f);
1003  const Packet alpha_9 = pset1<Packet>(4.37031012579801e-11f);
1004 
1005  // The monomial coefficients of the denominator polynomial (even).
1006  const Packet beta_0 = pset1<Packet>(9.93151921023180e-01f);
1007  const Packet beta_2 = pset1<Packet>(1.16817656904453e-01f);
1008  const Packet beta_4 = pset1<Packet>(1.70198817374094e-03f);
1009  const Packet beta_6 = pset1<Packet>(6.29106785017040e-06f);
1010  const Packet beta_8 = pset1<Packet>(5.76102136993427e-09f);
1011  const Packet beta_10 = pset1<Packet>(6.10247389755681e-13f);
1012 
1013  // Since the polynomials are odd/even, we need x^2.
1014  const Packet x2 = pmul(x, x);
1015 
1016  // Evaluate the numerator polynomial p.
1017  Packet p = pmadd(x2, alpha_9, alpha_7);
1018  p = pmadd(x2, p, alpha_5);
1019  p = pmadd(x2, p, alpha_3);
1020  p = pmadd(x2, p, alpha_1);
1021  p = pmul(x, p);
1022 
1023  // Evaluate the denominator polynomial q.
1024  Packet q = pmadd(x2, beta_10, beta_8);
1025  q = pmadd(x2, q, beta_6);
1026  q = pmadd(x2, q, beta_4);
1027  q = pmadd(x2, q, beta_2);
1028  q = pmadd(x2, q, beta_0);
1029  // Divide the numerator by the denominator and shift it up.
1030  const Packet logistic = padd(pdiv(p, q), pset1<Packet>(0.5f));
1031  if (EIGEN_PREDICT_FALSE(any_small)) {
1032  const Packet exponential = pexp(_x);
1033  return pselect(lt_mask, exponential, logistic);
1034  } else {
1035  return logistic;
1036  }
1037  }
1038 };
