60 SPDLOG_INLINE void registry::initialize_logger(std::shared_ptr<logger> new_logger)
61 {
62  std::lock_guard<std::mutex> lock(logger_map_mutex_);
63  new_logger->set_formatter(formatter_->clone());
64 
65  if (err_handler_)
66  {
67  new_logger->set_error_handler(err_handler_);
68  }
69 
70  new_logger->set_level(levels_.get(new_logger->name()));
71  new_logger->flush_on(flush_level_);
72 
73  if (backtrace_n_messages_ > 0)
74  {
75  new_logger->enable_backtrace(backtrace_n_messages_);
76  }
77 
78  if (automatic_registration_)
79  {
80  register_logger_(std::move(new_logger));
81  }
82 }

106 // set default logger.
107 // default logger is stored in default_logger_ (for faster retrieval) and in the loggers_ map.
108 SPDLOG_INLINE void registry::set_default_logger(std::shared_ptr<logger> new_default_logger)
109 {
110  std::lock_guard<std::mutex> lock(logger_map_mutex_);
111  // remove previous default logger from the map
112  if (default_logger_ != nullptr)
113  {
114  loggers_.erase(default_logger_->name());
115  }
116  if (new_default_logger != nullptr)
117  {
118  loggers_[new_default_logger->name()] = new_default_logger;
119  }
120  default_logger_ = std::move(new_default_logger);
121 }
