63 // Functions for sqrt.
64 // The EIGEN_FAST_MATH version uses the _mm_rsqrt_ps approximation and one step
65 // of Newton's method, at a cost of 1-2 bits of precision as opposed to the
66 // exact solution. It does not handle +inf, or denormalized numbers correctly.
67 // The main advantage of this approach is not just speed, but also the fact that
68 // it can be inlined and pipelined with other computations, further reducing its
69 // effective latency. This is similar to Quake3's fast inverse square root.
70 // For detail see here: http://www.beyond3d.com/content/articles/8/
71 template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
72 Packet4f psqrt<Packet4f>(const Packet4f& _x)
73 {
74  Packet4f half = pmul(_x, pset1<Packet4f>(.5f));
75  Packet4f denormal_mask = _mm_and_ps(
76  _mm_cmpge_ps(_x, _mm_setzero_ps()),
77  _mm_cmplt_ps(_x, pset1<Packet4f>((std::numeric_limits<float>::min)())));
78 
79  // Compute approximate reciprocal sqrt.
80  Packet4f x = _mm_rsqrt_ps(_x);
81  // Do a single step of Newton's iteration.
82  x = pmul(x, psub(pset1<Packet4f>(1.5f), pmul(half, pmul(x,x))));
83  // Flush results for denormals to zero.
84  return _mm_andnot_ps(denormal_mask, pmul(_x,x));
85 }

99 template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
100 Packet4f prsqrt<Packet4f>(const Packet4f& _x) {
101  _EIGEN_DECLARE_CONST_Packet4f(one_point_five, 1.5f);
102  _EIGEN_DECLARE_CONST_Packet4f(minus_half, -0.5f);
103  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(inf, 0x7f800000u);
104  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(flt_min, 0x00800000u);
105 
106  Packet4f neg_half = pmul(_x, p4f_minus_half);
107 
108  // Identity infinite, zero, negative and denormal arguments.
109  Packet4f lt_min_mask = _mm_cmplt_ps(_x, p4f_flt_min);
110  Packet4f inf_mask = _mm_cmpeq_ps(_x, p4f_inf);
111  Packet4f not_normal_finite_mask = _mm_or_ps(lt_min_mask, inf_mask);
112 
113  // Compute an approximate result using the rsqrt intrinsic.
114  Packet4f y_approx = _mm_rsqrt_ps(_x);
115 
116  // Do a single step of Newton-Raphson iteration to improve the approximation.
117  // This uses the formula y_{n+1} = y_n * (1.5 - y_n * (0.5 * x) * y_n).
118  // It is essential to evaluate the inner term like this because forming
119  // y_n^2 may over- or underflow.
120  Packet4f y_newton = pmul(
121  y_approx, pmadd(y_approx, pmul(neg_half, y_approx), p4f_one_point_five));
122 
123  // Select the result of the Newton-Raphson step for positive normal arguments.
124  // For other arguments, choose the output of the intrinsic. This will
125  // return rsqrt(+inf) = 0, rsqrt(x) = NaN if x < 0, and rsqrt(x) = +inf if
126  // x is zero or a positive denormalized float (equivalent to flushing positive
127  // denormalized inputs to zero).
128  return pselect<Packet4f>(not_normal_finite_mask, y_approx, y_newton);
129 }
