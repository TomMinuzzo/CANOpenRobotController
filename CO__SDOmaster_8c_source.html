111 /*
112  * Read received message from CAN module.
113  *
114  * Function will be called (by CAN receive interrupt) every time, when CAN
115  * message with correct identifier will be received. For more information and
116  * description of parameters see file CO_driver.h.
117  */
118 static void CO_SDOclient_receive(void *object, const CO_CANrxMsg_t *msg){
119  CO_SDOclient_t *SDO_C;
120 
121  SDO_C = (CO_SDOclient_t*)object; /* this is the correct pointer type of the first argument */
122 
123  /* verify message length and message overflow (previous message was not processed yet) */
124  if((msg->DLC == 8U) && (!SDO_C->CANrxNew) && (SDO_C->state != SDO_STATE_NOTDEFINED)){
125  if(SDO_C->state != SDO_STATE_BLOCKUPLOAD_INPROGRES) {
126  /* copy data and set 'new message' flag */
127  SDO_C->CANrxData[0] = msg->data[0];
128  SDO_C->CANrxData[1] = msg->data[1];
129  SDO_C->CANrxData[2] = msg->data[2];
130  SDO_C->CANrxData[3] = msg->data[3];
131  SDO_C->CANrxData[4] = msg->data[4];
132  SDO_C->CANrxData[5] = msg->data[5];
133  SDO_C->CANrxData[6] = msg->data[6];
134  SDO_C->CANrxData[7] = msg->data[7];
135 
136  SDO_C->CANrxNew = true;
137  }
138  else {
139  /* block upload, copy data directly */
140  uint8_t seqno;
141 
142  SDO_C->CANrxData[0] = msg->data[0];
143  seqno = SDO_C->CANrxData[0] & 0x7f;
144  SDO_C->timeoutTimer = 0;
145  SDO_C->timeoutTimerBLOCK = 0;
146 
147  /* check correct sequence number. */
148  if(seqno == (SDO_C->block_seqno + 1)) {
149  /* block_seqno is correct */
150  uint8_t i;
151 
152  SDO_C->block_seqno++;
153 
154  /* copy data */
155  for(i=1; i<8; i++) {
156  SDO_C->buffer[SDO_C->dataSizeTransfered++] = msg->data[i];
157  if(SDO_C->dataSizeTransfered >= SDO_C->bufferSize) {
158  /* buffer full, break reception */
159  SDO_C->state = SDO_STATE_BLOCKUPLOAD_SUB_END;
160  SDO_C->CANrxNew = true;
161  break;
162  }
163  }
164 
165  /* break reception if last segment or block sequence is too large */
166  if(((SDO_C->CANrxData[0] & 0x80U) == 0x80U) || (SDO_C->block_seqno >= SDO_C->block_blksize)) {
167  SDO_C->state = SDO_STATE_BLOCKUPLOAD_SUB_END;
168  SDO_C->CANrxNew = true;
169  }
170  }
171  else if((seqno == SDO_C->block_seqno) || (SDO_C->block_seqno == 0U)){
172  /* Ignore message, if it is duplicate or if sequence didn't started yet. */
173  }
174  else {
175  /* seqno is totally wrong, break reception. */
176  SDO_C->state = SDO_STATE_BLOCKUPLOAD_SUB_END;
177  SDO_C->CANrxNew = true;
178  }
179  }
180 
181  /* Optional signal to RTOS, which can resume task, which handles SDO client. */
182  if(SDO_C->CANrxNew && SDO_C->pFunctSignal != NULL) {
183  SDO_C->pFunctSignal();
184  }
185  }
186 }

242 CO_SDOclient_return_t CO_SDOclient_setup(
243  CO_SDOclient_t *SDO_C,
244  uint32_t COB_IDClientToServer,
245  uint32_t COB_IDServerToClient,
246  uint8_t nodeIDOfTheSDOServer)

248  uint32_t idCtoS, idStoC;
249  uint8_t idNode;
250 
251  /* verify parameters */
252  if(SDO_C == NULL || (COB_IDClientToServer&0x7FFFF800L) != 0 ||
253  (COB_IDServerToClient&0x7FFFF800L) != 0 || nodeIDOfTheSDOServer > 127)
254  {
255  return CO_SDOcli_wrongArguments;
256  }
257 
258  /* Configure object variables */
259  SDO_C->state = SDO_STATE_NOTDEFINED;
260  SDO_C->CANrxNew = false;
261 
262  /* setup Object Dictionary variables */
263  if((COB_IDClientToServer & 0x80000000L) != 0 || (COB_IDServerToClient & 0x80000000L) != 0 || nodeIDOfTheSDOServer == 0){
264  /* SDO is NOT used */
265  idCtoS = 0x80000000L;
266  idStoC = 0x80000000L;
267  idNode = 0;
268  }
269  else{
270  if(COB_IDClientToServer == 0 || COB_IDServerToClient == 0){
271  idCtoS = 0x600 + nodeIDOfTheSDOServer;
272  idStoC = 0x580 + nodeIDOfTheSDOServer;
273  }
274  else{
275  idCtoS = COB_IDClientToServer;
276  idStoC = COB_IDServerToClient;
277  }
278  idNode = nodeIDOfTheSDOServer;
279  }
280 
281  SDO_C->SDOClientPar->COB_IDClientToServer = idCtoS;
282  SDO_C->SDOClientPar->COB_IDServerToClient = idStoC;
283  SDO_C->SDOClientPar->nodeIDOfTheSDOServer = idNode;
284 
285  /* configure SDO client CAN reception, if differs. */
286  if(SDO_C->COB_IDClientToServerPrev != idCtoS || SDO_C->COB_IDServerToClientPrev != idStoC) {
287  CO_CANrxBufferInit(
288  SDO_C->CANdevRx, /* CAN device */
289  SDO_C->CANdevRxIdx, /* rx buffer index */
290  (uint16_t)idStoC, /* CAN identifier */
291  0x7FF, /* mask */
292  0, /* rtr */
293  (void*)SDO_C, /* object passed to receive function */
294  CO_SDOclient_receive); /* this function will process received message */
295 
296  /* configure SDO client CAN transmission */
297  SDO_C->CANtxBuff = CO_CANtxBufferInit(
298  SDO_C->CANdevTx, /* CAN device */
299  SDO_C->CANdevTxIdx, /* index of specific buffer inside CAN module */
300  (uint16_t)idCtoS, /* CAN identifier */
301  0, /* rtr */
302  8, /* number of data bytes */
303  0); /* synchronous message flag bit */
304 
305  SDO_C->COB_IDClientToServerPrev = idCtoS;
306  SDO_C->COB_IDServerToClientPrev = idStoC;
307  }
308 
309  return CO_SDOcli_ok_communicationEnd;
310 }

343 CO_SDOclient_return_t CO_SDOclientDownloadInitiate(
344  CO_SDOclient_t *SDO_C,
345  uint16_t index,
346  uint8_t subIndex,
347  uint8_t *dataTx,
348  uint32_t dataSize,
349  uint8_t blockEnable)
350 {
351  /* verify parameters */
352  if(SDO_C == NULL || dataTx == 0 || dataSize == 0) {
353  return CO_SDOcli_wrongArguments;
354  }
355 
356  /* save parameters */
357  SDO_C->buffer = dataTx;
358  SDO_C->bufferSize = dataSize;
359 
360  SDO_C->state = SDO_STATE_DOWNLOAD_INITIATE;
361 
362  /* prepare CAN tx message */
363  CO_SDOTxBufferClear(SDO_C);
364 
365  SDO_C->index = index;
366  SDO_C->subIndex = subIndex;
367  SDO_C->CANtxBuff->data[1] = index & 0xFF;
368  SDO_C->CANtxBuff->data[2] = index >> 8;
369  SDO_C->CANtxBuff->data[3] = subIndex;
370 
371  /* if nodeIDOfTheSDOServer == node-ID of this node, then exchange data with this node */
372  if(SDO_C->SDOClientPar->nodeIDOfTheSDOServer == SDO_C->SDO->nodeId){
373  return CO_SDOcli_ok_communicationEnd;
374  }
375 
376  if(dataSize <= 4){
377  uint16_t i;
378  /* expedited transfer */
379  SDO_C->CANtxBuff->data[0] = 0x23 | ((4-dataSize) << 2);
380 
381  /* copy data */
382  for(i=dataSize+3; i>=4; i--) SDO_C->CANtxBuff->data[i] = dataTx[i-4];
383  }
384  else if((SDO_C->bufferSize > SDO_C->pst) && blockEnable != 0){ /* BLOCK transfer */
385  /* set state of block transfer */
386  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_INITIATE;
387 
388  /* init HEAD of packet */
389  SDO_C->CANtxBuff->data[0] = CCS_DOWNLOAD_BLOCK<<5;
390  /* set flag for CRC */
391  SDO_C->CANtxBuff->data[0] |= 0x01<<2;
392 
393  /* size indicator: */
394  SDO_C->CANtxBuff->data[0] |= 0x01<<1;
395  /* set length of data */
396  SDO_C->CANtxBuff->data[4] = (uint8_t) dataSize;
397  SDO_C->CANtxBuff->data[5] = (uint8_t) (dataSize >> 8);
398  SDO_C->CANtxBuff->data[6] = (uint8_t) (dataSize >> 16);
399  SDO_C->CANtxBuff->data[7] = (uint8_t) (dataSize >> 24);
400 
401  }
402  else{
403  uint32_t len;
404  /* segmented transfer */
405  SDO_C->CANtxBuff->data[0] = 0x21;
406  len = dataSize;
407  CO_memcpySwap4(&SDO_C->CANtxBuff->data[4], &len);
408  }
409 
410  /* empty receive buffer, reset timeout timer and send message */
411  SDO_C->CANrxNew = false;
412  SDO_C->timeoutTimer = 0;
413  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
414 
415  return CO_SDOcli_ok_communicationEnd;
416 }

419 /******************************************************************************/
420 CO_SDOclient_return_t CO_SDOclientDownload(
421  CO_SDOclient_t *SDO_C,
422  uint16_t timeDifference_ms,
423  uint16_t SDOtimeoutTime,
424  uint32_t *pSDOabortCode)
425 {
426  CO_SDOclient_return_t ret = CO_SDOcli_waitingServerResponse;
427 
428  /* verify parameters */
429  if(SDO_C == NULL) {
430  return CO_SDOcli_wrongArguments;
431  }
432 
433  /* clear abort code */
434  *pSDOabortCode = CO_SDO_AB_NONE;
435 
436  /* if nodeIDOfTheSDOServer == node-ID of this node, then exchange data with this node */
437  if(SDO_C->SDO && SDO_C->SDOClientPar->nodeIDOfTheSDOServer == SDO_C->SDO->nodeId){
438  SDO_C->state = SDO_STATE_NOTDEFINED;
439  SDO_C->CANrxNew = false;
440 
441  /* If SDO server is busy return error */
442  if(SDO_C->SDO->state != 0){
443  return CO_SDOcli_endedWithClientAbort;
444  }
445 
446  /* init ODF_arg */
447  *pSDOabortCode = CO_SDO_initTransfer(SDO_C->SDO, SDO_C->index, SDO_C->subIndex);
448  if((*pSDOabortCode) != CO_SDO_AB_NONE){
449  return CO_SDOcli_endedWithServerAbort;
450  }
451 
452  /* set buffer */
453  SDO_C->SDO->ODF_arg.data = SDO_C->buffer;
454 
455  /* write data to the Object dictionary */
456  *pSDOabortCode = CO_SDO_writeOD(SDO_C->SDO, SDO_C->bufferSize);
457  if((*pSDOabortCode) != CO_SDO_AB_NONE){
458  return CO_SDOcli_endedWithServerAbort;
459  }
460 
461  return CO_SDOcli_ok_communicationEnd;
462  }
463 
464 
465 /* RX data ****************************************************************************************** */
466  if(SDO_C->CANrxNew){
467  uint8_t SCS = SDO_C->CANrxData[0]>>5; /* Client command specifier */
468 
469  /* ABORT */
470  if (SDO_C->CANrxData[0] == (SCS_ABORT<<5)){
471  SDO_C->state = SDO_STATE_NOTDEFINED;
472  CO_memcpySwap4(pSDOabortCode , &SDO_C->CANrxData[4]);
473  SDO_C->CANrxNew = false;
474  return CO_SDOcli_endedWithServerAbort;
475  }
476 
477  switch (SDO_C->state){
478 
479  case SDO_STATE_DOWNLOAD_INITIATE:{
480 
481  if (SCS == SCS_DOWNLOAD_INITIATED){
482  if(SDO_C->bufferSize <= 4){
483  /* expedited transfer */
484  SDO_C->state = SDO_STATE_NOTDEFINED;
485  SDO_C->CANrxNew = false;
486  return CO_SDOcli_ok_communicationEnd;
487  }
488  else{
489  /* segmented transfer - prepare the first segment */
490  SDO_C->bufferOffset = 0;
491  SDO_C->toggle =0;
492  SDO_C->state = SDO_STATE_DOWNLOAD_REQUEST;
493  }
494  }
495  else{
496  *pSDOabortCode = CO_SDO_AB_CMD;
497  SDO_C->state = SDO_STATE_ABORT;
498  }
499  break;
500  }
501 
502  case SDO_STATE_DOWNLOAD_RESPONSE:{
503 
504  if (SCS == SCS_DOWNLOAD_SEGMENT){
505  /* verify toggle bit */
506  if((SDO_C->CANrxData[0]&0x10) != (SDO_C->toggle<<4)){
507  *pSDOabortCode = CO_SDO_AB_TOGGLE_BIT;
508  SDO_C->state = SDO_STATE_ABORT;
509  break;
510  }
511  /* alternate toggle bit */
512  if (SDO_C->toggle ==0x00)
513  SDO_C->toggle =0x01;
514  else
515  SDO_C->toggle =0x00;
516 
517  /* is end of transfer? */
518  if(SDO_C->bufferOffset == SDO_C->bufferSize){
519  SDO_C->state = SDO_STATE_NOTDEFINED;
520  SDO_C->CANrxNew = false;
521  return CO_SDOcli_ok_communicationEnd;
522  }
523  SDO_C->state = SDO_STATE_DOWNLOAD_REQUEST;
524  }
525  else{
526  *pSDOabortCode = CO_SDO_AB_CMD;
527  SDO_C->state = SDO_STATE_ABORT;
528  }
529  break;
530  }
531 
532  case SDO_STATE_BLOCKDOWNLOAD_INITIATE:{ /* waiting on reply on block download initiated */
533  if (SCS == SCS_DOWNLOAD_BLOCK){
534  uint16_t IndexTmp;
535  IndexTmp = SDO_C->CANrxData[2];
536  IndexTmp = IndexTmp << 8;
537  IndexTmp |= SDO_C->CANrxData[1];
538 
539  if(IndexTmp != SDO_C->index || SDO_C->CANrxData[3] != SDO_C->subIndex) {
540  /* wrong index */
541  *pSDOabortCode = CO_SDO_AB_PRAM_INCOMPAT;
542  SDO_C->state = SDO_STATE_ABORT;
543  break;
544  }
545  /* set blksize */
546  SDO_C->block_blksize = SDO_C->CANrxData[4];
547 
548  SDO_C->block_seqno = 0;
549  SDO_C->bufferOffset = 0;
550  SDO_C->bufferOffsetACK = 0;
551  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_INPORGRES;
552 
553  break;
554  }
555  else{
556  *pSDOabortCode = CO_SDO_AB_CMD;
557  SDO_C->state = SDO_STATE_ABORT;
558  }
559  }
560 
561  case SDO_STATE_BLOCKDOWNLOAD_INPORGRES:
562  case SDO_STATE_BLOCKDOWNLOAD_BLOCK_ACK:{ /* waiting block ACK */
563  if (SCS == SCS_DOWNLOAD_BLOCK){
564  /* check server subcommand */
565  if((SDO_C->CANrxData[0] & 0x02) == 0){
566  /* wrong server sub command */
567  *pSDOabortCode = CO_SDO_AB_CMD;
568  SDO_C->state = SDO_STATE_ABORT;
569  break;
570  }
571  /* check number of segments */
572  if(SDO_C->CANrxData[1] != SDO_C->block_blksize){
573  /* NOT all segments transferred successfully */
574  SDO_C->bufferOffsetACK += SDO_C->CANrxData[1] * 7;
575  SDO_C->bufferOffset = SDO_C->bufferOffsetACK;
576  }
577  else{
578  SDO_C->bufferOffsetACK = SDO_C->bufferOffset;
579  }
580  /* set size of next block */
581  SDO_C->block_blksize = SDO_C->CANrxData[2];
582  SDO_C->block_seqno = 0;
583 
584  if(SDO_C->bufferOffset >= SDO_C->bufferSize)
585  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_CRC;
586  else
587  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_INPORGRES;
588  }
589  else{
590  *pSDOabortCode = CO_SDO_AB_CMD;
591  SDO_C->state = SDO_STATE_ABORT;
592  }
593  break;
594  }
595 
596  case SDO_STATE_BLOCKDOWNLOAD_CRC_ACK:{
597  if (SCS == SCS_DOWNLOAD_BLOCK){
598  if((SDO_C->CANrxData[0] & 0x01) == 0){
599  /* wrong server sub command */
600  *pSDOabortCode = CO_SDO_AB_CMD;
601  SDO_C->state = SDO_STATE_ABORT;
602  break;
603  }
604  /* SDO block download successfully transferred */
605  SDO_C->state = SDO_STATE_NOTDEFINED;
606  SDO_C->timeoutTimer = 0;
607  SDO_C->CANrxNew = false;
608  return CO_SDOcli_ok_communicationEnd;
609  }
610  else{
611  *pSDOabortCode = CO_SDO_AB_CMD;
612  SDO_C->state = SDO_STATE_ABORT;
613  break;
614  }
615  }
616 
617  default:{
618  *pSDOabortCode = CO_SDO_AB_CMD;
619  SDO_C->state = SDO_STATE_ABORT;
620  }
621  }
622  SDO_C->timeoutTimer = 0;
623  SDO_C->CANrxNew = false;
624  }
625 
626 /* TMO *********************************************************************************************** */
627  if(SDO_C->timeoutTimer < SDOtimeoutTime){
628  SDO_C->timeoutTimer += timeDifference_ms;
629  }
630  if(SDO_C->timeoutTimer >= SDOtimeoutTime){ /* communication TMO */
631  *pSDOabortCode = CO_SDO_AB_TIMEOUT;
632  CO_SDOclient_abort(SDO_C, *pSDOabortCode);
633  return CO_SDOcli_endedWithTimeout;
634  }
635 
636 /* TX data ******************************************************************************************* */
637  if(SDO_C->CANtxBuff->bufferFull) {
638  return CO_SDOcli_transmittBufferFull;
639  }
640 
641  CO_SDOTxBufferClear(SDO_C);
642  switch (SDO_C->state){
643  /* ABORT */
644  case SDO_STATE_ABORT:{
645  SDO_C->state = SDO_STATE_NOTDEFINED;
646  CO_SDOclient_abort (SDO_C, *pSDOabortCode);
647  ret = CO_SDOcli_endedWithClientAbort;
648  break;
649  }
650  /* SEGMENTED */
651  case SDO_STATE_DOWNLOAD_REQUEST:{
652  uint16_t i, j;
653  /* calculate length to be sent */
654  j = SDO_C->bufferSize - SDO_C->bufferOffset;
655  if(j > 7) j = 7;
656  /* fill data bytes */
657  for(i=0; i<j; i++)
658  SDO_C->CANtxBuff->data[i+1] = SDO_C->buffer[SDO_C->bufferOffset + i];
659 
660  for(; i<7; i++)
661  SDO_C->CANtxBuff->data[i+1] = 0;
662 
663  SDO_C->bufferOffset += j;
664  /* SDO command specifier */
665  SDO_C->CANtxBuff->data[0] = CCS_DOWNLOAD_SEGMENT | ((SDO_C->toggle)<<4) | ((7-j)<<1);
666  /* is end of transfer? */
667  if(SDO_C->bufferOffset == SDO_C->bufferSize){
668  SDO_C->CANtxBuff->data[0] |= 1;
669  }
670  /* Send next SDO message */
671  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
672  SDO_C->state = SDO_STATE_DOWNLOAD_RESPONSE;
673  break;
674  }
675 
676  /* BLOCK */
677  case SDO_STATE_BLOCKDOWNLOAD_INPORGRES:{
678  SDO_C->block_seqno += 1;
679  SDO_C->CANtxBuff->data[0] = SDO_C->block_seqno;
680 
681  if(SDO_C->block_seqno >= SDO_C->block_blksize){
682  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_BLOCK_ACK;
683  }
684  /* set data */
685  SDO_C->block_noData = 0;
686 
687  uint8_t i;
688  for(i = 1; i < 8; i++){
689  if(SDO_C->bufferOffset < SDO_C->bufferSize){
690  SDO_C->CANtxBuff->data[i] = *(SDO_C->buffer + SDO_C->bufferOffset);
691  }
692  else{
693  SDO_C->CANtxBuff->data[i] = 0;
694  SDO_C->block_noData += 1;
695  }
696 
697  SDO_C->bufferOffset += 1;
698  }
699 
700  if(SDO_C->bufferOffset >= SDO_C->bufferSize){
701  SDO_C->CANtxBuff->data[0] |= 0x80;
702  SDO_C->block_blksize = SDO_C->block_seqno;
703  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_BLOCK_ACK;
704  }
705 
706  /* tx data */
707  SDO_C->timeoutTimer = 0;
708  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
709 
710  break;
711  }
712 
713  case SDO_STATE_BLOCKDOWNLOAD_CRC:{
714  SDO_C->CANtxBuff->data[0] = (CCS_DOWNLOAD_BLOCK<<5) | (SDO_C->block_noData << 2) | 0x01;
715 
716  uint16_t tmp16;
717 
718  tmp16 = crc16_ccitt((unsigned char *)SDO_C->buffer, (unsigned int)SDO_C->bufferSize, 0);
719 
720  SDO_C->CANtxBuff->data[1] = (uint8_t) tmp16;
721  SDO_C->CANtxBuff->data[2] = (uint8_t) (tmp16>>8);
722 
723  /* set state */
724  SDO_C->state = SDO_STATE_BLOCKDOWNLOAD_CRC_ACK;
725  /* tx data */
726  SDO_C->timeoutTimer = 0;
727  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
728 
729  break;
730  }
731 
732  default:{
733  break;
734  }
735  }
736 
737  if(SDO_C->state == SDO_STATE_BLOCKDOWNLOAD_INPORGRES) {
738  ret = CO_SDOcli_blockDownldInProgress;
739  }
740 
741  return ret;
742 }

828  uint32_t tmp32;
829  CO_SDOclient_return_t ret = CO_SDOcli_waitingServerResponse;
830 
831  /* verify parameters */
832  if(SDO_C == NULL) {
833  return CO_SDOcli_wrongArguments;
834  }
835 
836  /* clear abort code */
837  *pSDOabortCode = CO_SDO_AB_NONE;
838 
839  /* if nodeIDOfTheSDOServer == node-ID of this node, then exchange data with this node */
840  if(SDO_C->SDO && SDO_C->SDOClientPar->nodeIDOfTheSDOServer == SDO_C->SDO->nodeId){
841  SDO_C->state = SDO_STATE_NOTDEFINED;
842  SDO_C->CANrxNew = false;
843 
844  /* If SDO server is busy return error */
845  if(SDO_C->SDO->state != 0){
846  *pSDOabortCode = CO_SDO_AB_DEVICE_INCOMPAT;
847  return CO_SDOcli_endedWithClientAbort;
848  }
849 
850  /* init ODF_arg */
851  *pSDOabortCode = CO_SDO_initTransfer(SDO_C->SDO, SDO_C->index, SDO_C->subIndex);
852  if((*pSDOabortCode) != CO_SDO_AB_NONE){
853  return CO_SDOcli_endedWithServerAbort;
854  }
855 
856  /* set buffer and length if domain */
857  SDO_C->SDO->ODF_arg.data = SDO_C->buffer;
858  if(SDO_C->SDO->ODF_arg.ODdataStorage == 0)
859  SDO_C->SDO->ODF_arg.dataLength = SDO_C->bufferSize;
860 
861  /* read data from the Object dictionary */
862  *pSDOabortCode = CO_SDO_readOD(SDO_C->SDO, SDO_C->bufferSize);
863  if((*pSDOabortCode) != CO_SDO_AB_NONE){
864  return CO_SDOcli_endedWithServerAbort;
865  }
866 
867  /* set data size */
868  *pDataSize = SDO_C->SDO->ODF_arg.dataLength;
869 
870  /* is SDO buffer too small */
871  if(SDO_C->SDO->ODF_arg.lastSegment == 0){
872  *pSDOabortCode = CO_SDO_AB_OUT_OF_MEM; /* Out of memory */
873  return CO_SDOcli_endedWithServerAbort;
874  }
875 
876  return CO_SDOcli_ok_communicationEnd;
877  }
878 
879 
880 /* RX data ******************************************************************************** */
881  if(SDO_C->CANrxNew){
882  uint8_t SCS = SDO_C->CANrxData[0]>>5; /* Client command specifier */
883 
884  /* ABORT */
885  if (SDO_C->CANrxData[0] == (SCS_ABORT<<5)){
886  SDO_C->state = SDO_STATE_NOTDEFINED;
887  SDO_C->CANrxNew = false;
888  CO_memcpySwap4(pSDOabortCode , &SDO_C->CANrxData[4]);
889  return CO_SDOcli_endedWithServerAbort;
890  }
891  switch (SDO_C->state){
892  case SDO_STATE_UPLOAD_INITIATED:{
893 
894 
895 
896  if (SCS == SCS_UPLOAD_INITIATE){
897  if(SDO_C->CANrxData[0] & 0x02){
898  uint8_t size;
899  /* Expedited transfer */
900  if(SDO_C->CANrxData[0] & 0x01)/* is size indicated */
901  size = 4 - ((SDO_C->CANrxData[0]>>2)&0x03); /* size */
902  else
903  size = 4;
904 
905  *pDataSize = size;
906 
907  /* copy data */
908  while(size--) SDO_C->buffer[size] = SDO_C->CANrxData[4+size];
909  SDO_C->state = SDO_STATE_NOTDEFINED;
910  SDO_C->CANrxNew = false;
911 
912  return CO_SDOcli_ok_communicationEnd;
913  }
914  else{
915  /* segmented transfer - prepare first segment */
916  SDO_C->bufferOffset = 0;
917  SDO_C->state = SDO_STATE_UPLOAD_REQUEST;
918 
919  SDO_C->toggle =0;
920  /* continue with segmented upload */
921  }
922  }
923  else{
924  *pSDOabortCode = CO_SDO_AB_CMD;
925  SDO_C->state = SDO_STATE_ABORT;
926  }
927  break;
928  }
929 
930  case SDO_STATE_UPLOAD_RESPONSE:{
931  if (SCS == SCS_UPLOAD_SEGMENT){
932  uint16_t size, i;
933  /* verify toggle bit */
934  if((SDO_C->CANrxData[0] &0x10) != (~SDO_C->toggle &0x10)){
935  *pSDOabortCode = CO_SDO_AB_TOGGLE_BIT;
936  SDO_C->state = SDO_STATE_ABORT;
937  break;
938  }
939  /* get size */
940  size = 7 - ((SDO_C->CANrxData[0]>>1)&0x07);
941  /* verify length */
942  if((SDO_C->bufferOffset + size) > SDO_C->bufferSize){
943  *pSDOabortCode = CO_SDO_AB_OUT_OF_MEM; /* Out of memory */
944  SDO_C->state = SDO_STATE_ABORT;
945  break;
946  }
947  /* copy data to buffer */
948  for(i=0; i<size; i++)
949  SDO_C->buffer[SDO_C->bufferOffset + i] = SDO_C->CANrxData[1 + i];
950  SDO_C->bufferOffset += size;
951  /* If no more segments to be uploaded, finish communication */
952  if(SDO_C->CANrxData[0] & 0x01){
953  *pDataSize = SDO_C->bufferOffset;
954  SDO_C->state = SDO_STATE_NOTDEFINED;
955  SDO_C->CANrxNew = false;
956  return CO_SDOcli_ok_communicationEnd;
957  }
958  /* set state */
959  SDO_C->state = SDO_STATE_UPLOAD_REQUEST;
960  break;
961  }
962  else{
963  *pSDOabortCode = CO_SDO_AB_CMD;
964  SDO_C->state = SDO_STATE_ABORT;
965  }
966  break;
967  }
968 
969  case SDO_STATE_BLOCKUPLOAD_INITIATE:{
970  if (SCS == SCS_UPLOAD_BLOCK){ /* block upload initiate response */
971 
972  SDO_C->state = SDO_STATE_BLOCKUPLOAD_INITIATE_ACK;
973 
974  /* SCR support */
975  if((SDO_C->CANrxData[0] & 0x04) != 0)
976  SDO_C->crcEnabled = 1; /* CRC suported */
977  else
978  SDO_C->crcEnabled = 0; /* CRC not suported */
979 
980  /* chech Index ans subnindex */
981  indexTmp = SDO_C->CANrxData[2];
982  indexTmp = indexTmp << 8;
983  indexTmp |= SDO_C->CANrxData[1];
984 
985  if(indexTmp != SDO_C->index || SDO_C->CANrxData[3] != SDO_C->subIndex){
986  *pSDOabortCode = CO_SDO_AB_PRAM_INCOMPAT;
987  SDO_C->state = SDO_STATE_ABORT;
988  }
989 
990  /* set length */
991  if(SDO_C->CANrxData[0]&0x02){
992  uint32_t len;
993  CO_memcpySwap4(&len, &SDO_C->CANrxData[4]);
994  SDO_C->dataSize = len;
995  }
996  else{
997  SDO_C->dataSize = 0;
998  }
999 
1000  /* check available buffer size */
1001  if (SDO_C->dataSize > SDO_C->bufferSize){
1002  *pSDOabortCode = CO_SDO_AB_OUT_OF_MEM;
1003  SDO_C->state = SDO_STATE_ABORT;
1004  }
1005 
1006  SDO_C->dataSizeTransfered =0;
1007  }
1008  else if (SCS == SCS_UPLOAD_INITIATE){ /* switch to regular segmented transfer */
1009  if(SDO_C->CANrxData[0] & 0x02){
1010  uint8_t size;
1011  /* Expedited transfer */
1012  if(SDO_C->CANrxData[0] & 0x01)/* is size indicated */
1013  size = 4 - ((SDO_C->CANrxData[0]>>2)&0x03); /* size */
1014  else
1015  size = 4;
1016 
1017  *pDataSize = size;
1018 
1019  /* copy data */
1020  while(size--) SDO_C->buffer[size] = SDO_C->CANrxData[4+size];
1021  SDO_C->state = SDO_STATE_NOTDEFINED;
1022  SDO_C->CANrxNew = false;
1023 
1024  return CO_SDOcli_ok_communicationEnd;
1025  }
1026  else{
1027  /* segmented transfer - prepare first segment */
1028  SDO_C->bufferOffset = 0;
1029  SDO_C->state = SDO_STATE_UPLOAD_REQUEST;
1030 
1031  SDO_C->toggle =0;
1032  /* continue with segmented upload */
1033  }
1034  }
1035  else{ /* unknown SCS */
1036  *pSDOabortCode = CO_SDO_AB_CMD;
1037  SDO_C->state = SDO_STATE_ABORT;
1038  }
1039  break;
1040  }
1041 
1042  case SDO_STATE_BLOCKUPLOAD_INPROGRES:{
1043  /* data are copied directly in receive function */
1044  break;
1045  }
1046 
1047  case SDO_STATE_BLOCKUPLOAD_SUB_END:{
1048  /* data was copied by receive function, sub-block is finished */
1049  /* Is last segment? */
1050  if(SDO_C->CANrxData[0] & 0x80) {
1051  /* Is data size indicated and wrong? */
1052  if((SDO_C->dataSize != 0) && (SDO_C->dataSize > SDO_C->dataSizeTransfered)) {
1053  *pSDOabortCode = CO_SDO_AB_TYPE_MISMATCH;
1054  SDO_C->state = SDO_STATE_ABORT;
1055  }
1056  else {
1057  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_ACK_LAST;
1058  }
1059  }
1060  else {
1061  /* Is SDO buffer overflow? */
1062  if(SDO_C->dataSizeTransfered >= SDO_C->bufferSize) {
1063  *pSDOabortCode = CO_SDO_AB_OUT_OF_MEM;
1064  SDO_C->state = SDO_STATE_ABORT;
1065  }
1066  else {
1067  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_ACK;
1068  }
1069  }
1070  break;
1071  }
1072 
1073  case SDO_STATE_BLOCKUPLOAD_BLOCK_CRC:{
1074  if (SCS == SCS_UPLOAD_BLOCK){
1075  tmp32 = ((SDO_C->CANrxData[0]>>2) & 0x07);
1076  SDO_C->dataSizeTransfered -= tmp32;
1077 
1078  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_END;
1079  if (SDO_C->crcEnabled){
1080  uint16_t tmp16;
1081  CO_memcpySwap2(&tmp16, &SDO_C->CANrxData[1]);
1082 
1083  if (tmp16 != crc16_ccitt((unsigned char *)SDO_C->buffer, (unsigned int)SDO_C->dataSizeTransfered, 0)){
1084  *pSDOabortCode = CO_SDO_AB_CRC;
1085  SDO_C->state = SDO_STATE_ABORT;
1086  }
1087  else{
1088  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_END;
1089  }
1090  }
1091  else{
1092  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_END;
1093  }
1094 
1095  }
1096  else{
1097  *pSDOabortCode = CO_SDO_AB_GENERAL;
1098  SDO_C->state = SDO_STATE_ABORT;
1099  }
1100  break;
1101  }
1102 
1103  default:{
1104  *pSDOabortCode = CO_SDO_AB_CMD;
1105  SDO_C->state = SDO_STATE_ABORT;
1106  }
1107  }
1108  SDO_C->timeoutTimer = 0;
1109  SDO_C->CANrxNew = false;
1110  }
1111 
1112 /* TMO *************************************************************************************************** */
1113  if(SDO_C->timeoutTimer < SDOtimeoutTime){
1114  SDO_C->timeoutTimer += timeDifference_ms;
1115  if (SDO_C->state == SDO_STATE_BLOCKUPLOAD_INPROGRES)
1116  SDO_C->timeoutTimerBLOCK += timeDifference_ms;
1117  }
1118  if(SDO_C->timeoutTimer >= SDOtimeoutTime){ /* communication TMO */
1119  *pSDOabortCode = CO_SDO_AB_TIMEOUT;
1120  CO_SDOclient_abort(SDO_C, *pSDOabortCode);
1121  return CO_SDOcli_endedWithTimeout;
1122  }
1123  if(SDO_C->timeoutTimerBLOCK >= (SDOtimeoutTime/2)){ /* block TMO */
1124  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_ACK;
1125  }
1126 
1127 
1128 /* TX data ******************************************************************************** */
1129  if(SDO_C->CANtxBuff->bufferFull) {
1130  return CO_SDOcli_transmittBufferFull;
1131  }
1132 
1133  CO_SDOTxBufferClear(SDO_C);
1134  switch (SDO_C->state){
1135  case SDO_STATE_ABORT:{
1136  SDO_C->state = SDO_STATE_NOTDEFINED;
1137  CO_SDOclient_abort (SDO_C, *pSDOabortCode);
1138  ret = CO_SDOcli_endedWithClientAbort;
1139  break;
1140  }
1141 
1142  /* SEGMENTED UPLOAD */
1143  case SDO_STATE_UPLOAD_REQUEST:{
1144  SDO_C->CANtxBuff->data[0] = (CCS_UPLOAD_SEGMENT<<5) | (SDO_C->toggle & 0x10);
1145  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
1146 
1147  SDO_C->state = SDO_STATE_UPLOAD_RESPONSE;
1148  SDO_C->toggle = ~SDO_C->toggle;
1149 
1150  break;;
1151  }
1152  /* BLOCK */
1153  case SDO_STATE_BLOCKUPLOAD_INITIATE_ACK:{
1154  SDO_C->timeoutTimerBLOCK = 0;
1155  SDO_C->block_seqno = 0;
1156  SDO_C->state = SDO_STATE_BLOCKUPLOAD_INPROGRES;
1157 
1158  /* header */
1159  SDO_C->CANtxBuff->data[0] = (CCS_UPLOAD_BLOCK<<5) | 0x03;
1160  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
1161 
1162  break;
1163  }
1164 
1165  case SDO_STATE_BLOCKUPLOAD_BLOCK_ACK_LAST:{
1166  /* header */
1167  SDO_C->CANtxBuff->data[0] = (CCS_UPLOAD_BLOCK<<5) | 0x02;
1168  SDO_C->CANtxBuff->data[1] = SDO_C->block_seqno;
1169 
1170  SDO_C->block_seqno = 0;
1171  SDO_C->timeoutTimerBLOCK = 0;
1172 
1173  SDO_C->state = SDO_STATE_BLOCKUPLOAD_BLOCK_CRC;
1174 
1175  SDO_C->CANtxBuff->data[2] = SDO_C->block_blksize;
1176  CO_CANsend(SDO_C->CANdevTx, SDO_C->CANtxBuff);
1177 
1178  break;
1179  }
1180
