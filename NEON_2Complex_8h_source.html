139 template<> EIGEN_STRONG_INLINE Packet1cf pmul<Packet1cf>(const Packet1cf& a, const Packet1cf& b)
140 {
141  Packet2f v1, v2;
142 
143  // Get the real values of a | a1_re | a1_re |
144  v1 = vdup_lane_f32(a.v, 0);
145  // Get the imag values of a | a1_im | a1_im |
146  v2 = vdup_lane_f32(a.v, 1);
147  // Multiply the real a with b
148  v1 = vmul_f32(v1, b.v);
149  // Multiply the imag a with b
150  v2 = vmul_f32(v2, b.v);
151  // Conjugate v2
152  v2 = vreinterpret_f32_u32(veor_u32(vreinterpret_u32_f32(v2), p2ui_CONJ_XOR()));
153  // Swap real/imag elements in v2.
154  v2 = vrev64_f32(v2);
155  // Add and return the result
156  return Packet1cf(vadd_f32(v1, v2));
157 }

178 template<> EIGEN_STRONG_INLINE Packet1cf pcmp_eq(const Packet1cf& a, const Packet1cf& b)
179 {
180  // Compare real and imaginary parts of a and b to get the mask vector:
181  // [re(a[0])==re(b[0]), im(a[0])==im(b[0])]
182  Packet2f eq = pcmp_eq<Packet2f>(a.v, b.v);
183  // Swap real/imag elements in the mask in to get:
184  // [im(a[0])==im(b[0]), re(a[0])==re(b[0])]
185  Packet2f eq_swapped = vrev64_f32(eq);
186  // Return re(a)==re(b) && im(a)==im(b) by computing bitwise AND of eq and eq_swapped
187  return Packet1cf(pand<Packet2f>(eq, eq_swapped));
188 }

514 {
515  Packet2d v1, v2;
516 
517  // Get the real values of a
518  v1 = vdupq_lane_f64(vget_low_f64(a.v), 0);
519  // Get the imag values of a
520  v2 = vdupq_lane_f64(vget_high_f64(a.v), 0);
521  // Multiply the real a with b
522  v1 = vmulq_f64(v1, b.v);
523  // Multiply the imag a with b
524  v2 = vmulq_f64(v2, b.v);
525  // Conjugate v2
526  v2 = vreinterpretq_f64_u64(veorq_u64(vreinterpretq_u64_f64(v2), p2ul_CONJ_XOR));
527  // Swap real/imag elements in v2.
528  v2 = preverse<Packet2d>(v2);
529  // Add and return the result
