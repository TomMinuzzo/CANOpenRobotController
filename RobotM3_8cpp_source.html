280 Vector3d RobotM3::inverseKinematic(Vector3d X) {
281  Vector3d q;
282 
283  float *L = LinkLengths;
284 
285  //Check accessible workspace
286  double normX = X.norm();
287  if ((L[4] < L[2] && normX < L[2] - L[4]) || (L[4] > L[2] && normX < sqrt(L[4] * L[4] - L[2] * L[2])) || normX > (L[2] + L[4] + L[0]) || X[0] > 0) {
288  std::cout /*cerr is banned*/ << "RobotM3::inverseKinematic() error: Point not accessible. NaN returned." << std::endl;
289  q[0] = q[1] = q[2] = nan("");
290  return q;
291  }
292 
293  //Compute first joint
294  q[0] = -atan2(X[1], -X[0]);
295 
296  //Project onto parallel mechanism plane
297  Vector3d tmpX;
298  if (X[0] > 0) {
299  //should never happen as outside of workspace...
300  tmpX[0] = sqrt(X[0] * X[0] + X[1] * X[1]);
301  } else {
302  tmpX[0] = -sqrt(X[0] * X[0] + X[1] * X[1]);
303  }
304  //Remove offset along -x
305  tmpX[0] = tmpX[0] + L[0];
306  tmpX[1] = X[1];
307  tmpX[2] = X[2];
308 
309  //Calculate joints 2 and 3
310  double beta = acos((L[2] * L[2] + L[4] * L[4] - tmpX[0] * tmpX[0] - tmpX[2] * tmpX[2]) / (2. * (L[2] * L[4])));
311  q[1] = acos(L[4] * sin(beta) / sqrt(tmpX[0] * tmpX[0] + tmpX[2] * tmpX[2])) - atan2(tmpX[2], -tmpX[0]);
312  q[2] = M_PI / 2. + q[1] - beta;
313 
314  return q;
315 }
316 Matrix3d RobotM3::J() {
317  Matrix3d J;
318  Vector3d q;
319  for (unsigned int i = 0; i < 3; i++) {
320  q(i) = ((JointM3 *)joints[i])->getPosition();
321  }
322 
323  float *L = LinkLengths;
324 
325  //Pre calculate factors for optimisation
326  double F1 = (L[1] + L[3]) * sin(q[2]);
327  double F2 = -L[2] * cos(q[1]);
328  double F3 = (L[1] + L[3]) * cos(q[2]) + L[2] * sin(q[1]) + L[0];
329 
330  //Jacobian matrix elements
331  J(0, 0) = F3 * sin(q[0]);
332  J(0, 1) = F2 * cos(q[0]);
333  J(0, 2) = F1 * cos(q[0]);
334 
335  J(1, 0) = -F3 * cos(q[0]);
336  J(1, 1) = F2 * sin(q[0]);
337  J(1, 2) = F1 * sin(q[0]);
338 
339  J(2, 0) = 0;
340  J(2, 1) = -L[2] * sin(q[1]);
341  J(2, 2) = -(L[1] + L[3]) * cos(q[2]);
342 
343  return J;
344 }

346 Vector3d RobotM3::calculateGravityTorques() {
347  Vector3d tau_g;
348 
349  //For convenience
350  float *L = LinkLengths;
351  float *M = LinkMasses;
352 
353  float g = 9.81; //Gravitational constant: remember to change it if using the robot on the Moon or another planet
354 
355  //Get current configuration
356  Vector3d q;
357  for (unsigned int i = 0; i < 3; i++) {
358  q(i) = ((JointM3 *)joints[i])->getPosition();
359  }
360 
361  //Calculate gravitational torques
362  tau_g[0] = 0;
363  tau_g[1] = -L[2] / 2.0f * sin(q[1]) * (M[1] + M[2] + M[3] + M[4]) * g;
364  tau_g[2] = -(L[1] / 2.0f * (M[0] + M[3]) + L[1] * M[2] + L[4] / 2.0f * M[4]) * cos(q[2]) * g;
365 
366  return tau_g;
367 }
