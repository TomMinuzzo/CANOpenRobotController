43 static CO_ReturnError_t setFilters(CO_CANmodule_t *CANmodule){
44  CO_ReturnError_t ret = CO_ERROR_NO;
45 
46  if(CANmodule->useCANrxFilters){
47  int nFiltersIn, nFiltersOut;
48  struct can_filter *filtersOut;
49 
50  nFiltersIn = CANmodule->rxSize;
51  nFiltersOut = 0;
52  filtersOut = (struct can_filter *) calloc(nFiltersIn, sizeof(struct can_filter));
53 
54  if(filtersOut == NULL){
55  ret = CO_ERROR_OUT_OF_MEMORY;
56  }else{
57  int i;
58  int idZeroCnt = 0;
59 
60  /* Copy filterIn to filtersOut. Accept only first filter with
61  * can_id=0, omit others. */
62  for(i=0; i<nFiltersIn; i++){
63  struct can_filter *fin;
64 
65  fin = &CANmodule->filter[i];
66  if(fin->can_id == 0){
67  idZeroCnt++;
68  }
69  if(fin->can_id != 0 || idZeroCnt == 1){
70  struct can_filter *fout;
71 
72  fout = &filtersOut[nFiltersOut++];
73  fout->can_id = fin->can_id;
74  fout->can_mask = fin->can_mask;
75  }
76  }
77 
78  if(setsockopt(CANmodule->fd, SOL_CAN_RAW, CAN_RAW_FILTER,
79  filtersOut, sizeof(struct can_filter) * nFiltersOut) != 0)
80  {
81  ret = CO_ERROR_ILLEGAL_ARGUMENT;
82  }
83 
84  free(filtersOut);
85  }
86  }else{
87  /* Use one socketCAN filter, match any CAN address, including extended and rtr. */
88  CANmodule->filter[0].can_id = 0;
89  CANmodule->filter[0].can_mask = 0;
90  if(setsockopt(CANmodule->fd, SOL_CAN_RAW, CAN_RAW_FILTER,
91  &CANmodule->filter[0], sizeof(struct can_filter)) != 0)
92  {
93  ret = CO_ERROR_ILLEGAL_ARGUMENT;
94  }
95  }
96 
97  return ret;
98 }

117 CO_ReturnError_t CO_CANmodule_init(
118  CO_CANmodule_t *CANmodule,
119  int32_t CANbaseAddress,
120  CO_CANrx_t rxArray[],
121  uint16_t rxSize,
122  CO_CANtx_t txArray[],
123  uint16_t txSize,
124  uint16_t CANbitRate)
125 {
126  CO_ReturnError_t ret = CO_ERROR_NO;
127  uint16_t i;
128 
129  /* verify arguments */
130  if(CANmodule==NULL || CANbaseAddress==0 || rxArray==NULL || txArray==NULL){
131  ret = CO_ERROR_ILLEGAL_ARGUMENT;
132  }
133 
134  /* Configure object variables */
135  if(ret == CO_ERROR_NO){
136  CANmodule->CANbaseAddress = CANbaseAddress;
137  CANmodule->rxArray = rxArray;
138  CANmodule->rxSize = rxSize;
139  CANmodule->txArray = txArray;
140  CANmodule->txSize = txSize;
141  CANmodule->CANnormal = false;
142  CANmodule->useCANrxFilters = true;
143  CANmodule->bufferInhibitFlag = false;
144  CANmodule->firstCANtxMessage = true;
145  CANmodule->error = 0;
146  CANmodule->CANtxCount = 0U;
147  CANmodule->errOld = 0U;
148  CANmodule->em = NULL;
149 
150 #ifdef CO_LOG_CAN_MESSAGES
151  CANmodule->useCANrxFilters = false;
152 #endif
153 
154  for(i=0U; i<rxSize; i++){
155  rxArray[i].ident = 0U;
156  rxArray[i].pFunct = NULL;
157  }
158  for(i=0U; i<txSize; i++){
159  txArray[i].bufferFull = false;
160  }
161  }
162 
163  /* First time only configuration */
164  if(ret == CO_ERROR_NO && CANmodule->wasConfigured == 0){
165  struct sockaddr_can sockAddr;
166 
167  CANmodule->wasConfigured = 1;
168 
169  /* Create and bind socket */
170  CANmodule->fd = socket(AF_CAN, SOCK_RAW, CAN_RAW);
171  if(CANmodule->fd < 0){
172  ret = CO_ERROR_ILLEGAL_ARGUMENT;
173  }else{
174  sockAddr.can_family = AF_CAN;
175  sockAddr.can_ifindex = CANbaseAddress;
176  if(bind(CANmodule->fd, (struct sockaddr*)&sockAddr, sizeof(sockAddr)) != 0){
177  ret = CO_ERROR_ILLEGAL_ARGUMENT;
178  }
179  }
180 
181  /* allocate memory for filter array */
182  if(ret == CO_ERROR_NO){
183  CANmodule->filter = (struct can_filter *) calloc(rxSize, sizeof(struct can_filter));
184  if(CANmodule->filter == NULL){
185  ret = CO_ERROR_OUT_OF_MEMORY;
186  }
187  }
188  }
189 
190  /* Additional check. */
191  if(ret == CO_ERROR_NO && CANmodule->filter == NULL){
192  ret = CO_ERROR_ILLEGAL_ARGUMENT;
193  }
194 
195  /* Configure CAN module hardware filters */
196  if(ret == CO_ERROR_NO && CANmodule->useCANrxFilters){
197  /* Match filter, standard 11 bit CAN address only, no rtr */
198  for(i=0U; i<rxSize; i++){
199  CANmodule->filter[i].can_id = 0;
200  CANmodule->filter[i].can_mask = CAN_SFF_MASK | CAN_EFF_FLAG | CAN_RTR_FLAG;
201  }
202  }
203 
204  /* close CAN module filters for now. */
205  if(ret == CO_ERROR_NO){
206  setsockopt(CANmodule->fd, SOL_CAN_RAW, CAN_RAW_FILTER, NULL, 0);
207  }
208 
209  return ret;
210 }

228 CO_ReturnError_t CO_CANrxBufferInit(
229  CO_CANmodule_t *CANmodule,
230  uint16_t index,
231  uint16_t ident,
232  uint16_t mask,
233  bool_t rtr,
234  void *object,
235  void (*pFunct)(void *object, const CO_CANrxMsg_t *message))
236 {
237  CO_ReturnError_t ret = CO_ERROR_NO;
238 
239  if((CANmodule!=NULL) && (object!=NULL) && (pFunct!=NULL) &&
240  (CANmodule->filter!=NULL) && (index < CANmodule->rxSize)){
241  /* buffer, which will be configured */
242  CO_CANrx_t *buffer = &CANmodule->rxArray[index];
243 
244  /* Configure object variables */
245  buffer->object = object;
246  buffer->pFunct = pFunct;
247 
248  /* Configure CAN identifier and CAN mask, bit aligned with CAN module. */
249  buffer->ident = ident & CAN_SFF_MASK;
250  if(rtr){
251  buffer->ident |= CAN_RTR_FLAG;
252  }
253  buffer->mask = (mask & CAN_SFF_MASK) | CAN_EFF_FLAG | CAN_RTR_FLAG;
254 
255  /* Set CAN hardware module filter and mask. */
256  if(CANmodule->useCANrxFilters){
257  CANmodule->filter[index].can_id = buffer->ident;
258  CANmodule->filter[index].can_mask = buffer->mask;
259  if(CANmodule->CANnormal){
260  ret = setFilters(CANmodule);
261  }
262  }
263  }
264  else{
265  ret = CO_ERROR_ILLEGAL_ARGUMENT;
266  }
267 
268  return ret;
269 }
