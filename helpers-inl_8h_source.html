46 inline std::pair<std::string, std::string> extract_kv_(char sep, const std::string &str)
47 {
48  auto n = str.find(sep);
49  std::string k, v;
50  if (n == std::string::npos)
51  {
52  v = str;
53  }
54  else
55  {
56  k = str.substr(0, n);
57  v = str.substr(n + 1);
58  }
59  return std::make_pair(trim_(k), trim_(v));

62 // return vector of key/value pairs from sequence of "K1=V1,K2=V2,.."
63 // "a=AAA,b=BBB,c=CCC,.." => {("a","AAA"),("b","BBB"),("c", "CCC"),...}
64 inline std::unordered_map<std::string, std::string> extract_key_vals_(const std::string &str)
65 {
66  std::string token;
67  std::istringstream token_stream(str);
68  std::unordered_map<std::string, std::string> rv{};
69  while (std::getline(token_stream, token, ','))
70  {
71  if (token.empty())
72  {
73  continue;
74  }
75  auto kv = extract_kv_('=', token);
76  rv[kv.first] = kv.second;
77  }
78  return rv;
79 }

81 SPDLOG_INLINE log_levels extract_levels(const std::string &input)
