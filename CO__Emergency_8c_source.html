59  CO_EMpr_t *emPr;
60  uint8_t value;
61  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
62

64  value = ODF_arg->data[0];
65

82  if(ODF_arg->subIndex == 0U){

84  emPr->preDefErrNoOfErrors = 0U;
85  }
86  else{
87  ret = CO_SDO_AB_INVALID_VALUE;
88  }
89  }
90  else{
91  ret = CO_SDO_AB_READONLY;
92  }
93  }

122 /******************************************************************************/
123 CO_ReturnError_t CO_EM_init(
124  CO_EM_t *em,
125  CO_EMpr_t *emPr,
126  CO_SDO_t *SDO,
127  uint8_t *errorStatusBits,
128  uint8_t errorStatusBitsSize,
129  uint8_t *errorRegister,
130  uint32_t *preDefErr,
131  uint8_t preDefErrSize,
132  CO_CANmodule_t *CANdev,
133  uint16_t CANdevTxIdx,
134  uint16_t CANidTxEM)
135 {
136  uint8_t i;
137 
138  /* verify arguments */
139  if(em==NULL || emPr==NULL || SDO==NULL || errorStatusBits==NULL ||
140  errorStatusBitsSize<6U || errorRegister==NULL || preDefErr==NULL || CANdev==NULL){
141  return CO_ERROR_ILLEGAL_ARGUMENT;
142  }
143 
144  /* Configure object variables */
145  em->errorStatusBits = errorStatusBits;
146  em->errorStatusBitsSize = errorStatusBitsSize;
147  em->bufEnd = em->buf + (CO_EM_INTERNAL_BUFFER_SIZE * 8);
148  em->bufWritePtr = em->buf;
149  em->bufReadPtr = em->buf;
150  em->bufFull = 0U;
151  em->wrongErrorReport = 0U;
152  em->pFunctSignal = NULL;
153  emPr->em = em;
154  emPr->errorRegister = errorRegister;
155  emPr->preDefErr = preDefErr;
156  emPr->preDefErrSize = preDefErrSize;
157  emPr->preDefErrNoOfErrors = 0U;
158  emPr->inhibitEmTimer = 0U;
159 
160  /* clear error status bits */
161  for(i=0U; i<errorStatusBitsSize; i++){
162  em->errorStatusBits[i] = 0U;
163  }
164 
165  /* Configure Object dictionary entry at index 0x1003 and 0x1014 */
166  CO_OD_configure(SDO, OD_H1003_PREDEF_ERR_FIELD, CO_ODF_1003, (void*)emPr, 0, 0U);
167  CO_OD_configure(SDO, OD_H1014_COBID_EMERGENCY, CO_ODF_1014, (void*)&SDO->nodeId, 0, 0U);
168 
169  /* configure emergency message CAN transmission */
170  emPr->CANdev = CANdev;
171  emPr->CANdev->em = (void*)em; /* update pointer inside CAN device. */
172  emPr->CANtxBuff = CO_CANtxBufferInit(
173  CANdev, /* CAN device */
174  CANdevTxIdx, /* index of specific buffer inside CAN module */
175  CANidTxEM, /* CAN identifier */
176  0, /* rtr */
177  8U, /* number of data bytes */
178  0); /* synchronous message flag bit */
179 
180  return CO_ERROR_NO;
181 }

195 /******************************************************************************/
196 void CO_EM_process(
197  CO_EMpr_t *emPr,
198  bool_t NMTisPreOrOperational,
199  uint16_t timeDifference_100us,
200  uint16_t emInhTime)
201 {
202 
203  CO_EM_t *em = emPr->em;
204  uint8_t errorRegister;
205 
206  /* verify errors from driver and other */
207  CO_CANverifyErrors(emPr->CANdev);
208  if(em->wrongErrorReport != 0U){
209  CO_errorReport(em, CO_EM_WRONG_ERROR_REPORT, CO_EMC_SOFTWARE_INTERNAL, (uint32_t)em->wrongErrorReport);
210  em->wrongErrorReport = 0U;
211  }
212 
213 
214  /* calculate Error register */
215  errorRegister = 0U;
216  /* generic error */
217  if(em->errorStatusBits[5]){
218  errorRegister |= CO_ERR_REG_GENERIC_ERR;
219  }
220  /* communication error (overrun, error state) */
221  if(em->errorStatusBits[2] || em->errorStatusBits[3]){
222  errorRegister |= CO_ERR_REG_COMM_ERR;
223  }
224  *emPr->errorRegister = (*emPr->errorRegister & 0xEEU) | errorRegister;
225 
226  /* inhibit time */
227  if(emPr->inhibitEmTimer < emInhTime){
228  emPr->inhibitEmTimer += timeDifference_100us;
229  }
230 
231  /* send Emergency message. */
232  if( NMTisPreOrOperational &&
233  !emPr->CANtxBuff->bufferFull &&
234  emPr->inhibitEmTimer >= emInhTime &&
235  (em->bufReadPtr != em->bufWritePtr || em->bufFull))
236  {
237  uint32_t preDEF; /* preDefinedErrorField */
238 
239  /* add error register */
240  em->bufReadPtr[2] = *emPr->errorRegister;
241 
242  /* copy data to CAN emergency message */
243  CO_memcpy(emPr->CANtxBuff->data, em->bufReadPtr, 8U);
244  CO_memcpy((uint8_t*)&preDEF, em->bufReadPtr, 4U);
245  em->bufReadPtr += 8;
246 
247  /* Update read buffer pointer and reset inhibit timer */
248  if(em->bufReadPtr == em->bufEnd){
249  em->bufReadPtr = em->buf;
250  }
251  emPr->inhibitEmTimer = 0U;
252 
253  /* verify message buffer overflow, then clear full flag */
254  if(em->bufFull == 2U){
255  em->bufFull = 0U; /* will be updated below */
256  CO_errorReport(em, CO_EM_EMERGENCY_BUFFER_FULL, CO_EMC_GENERIC, 0U);
257  }
258  else{
259  em->bufFull = 0;
260  }
261 
262  /* write to 'pre-defined error field' (object dictionary, index 0x1003) */
263  if(emPr->preDefErr){
264  uint8_t i;
265 
266  if(emPr->preDefErrNoOfErrors < emPr->preDefErrSize)
267  emPr->preDefErrNoOfErrors++;
268  for(i=emPr->preDefErrNoOfErrors-1; i>0; i--)
269  emPr->preDefErr[i] = emPr->preDefErr[i-1];
270  emPr->preDefErr[0] = preDEF;
271  }
272 
273  /* send CAN message */
274  CO_CANsend(emPr->CANdev, emPr->CANtxBuff);
275  }
276 
277  return;
278 }

281 /******************************************************************************/
282 void CO_errorReport(CO_EM_t *em, const uint8_t errorBit, const uint16_t errorCode, const uint32_t infoCode){
283  uint8_t index = errorBit >> 3;
284  uint8_t bitmask = 1 << (errorBit & 0x7);
285  uint8_t *errorStatusBits = 0;
286  bool_t sendEmergency = true;
287 
288  if(em == NULL){
289  sendEmergency = false;
290  }
291  else if(index >= em->errorStatusBitsSize){
292  /* if errorBit value not supported, send emergency 'CO_EM_WRONG_ERROR_REPORT' */
293  em->wrongErrorReport = errorBit;
294  sendEmergency = false;
295  }
296  else{
297  errorStatusBits = &em->errorStatusBits[index];
298  /* if error was already reported, do nothing */
299  if((*errorStatusBits & bitmask) != 0){
300  sendEmergency = false;
301  }
302  }
303 
304  if(sendEmergency){
305  /* set error bit */
306  if(errorBit){
307  /* any error except NO_ERROR */
308  *errorStatusBits |= bitmask;
309  }
310 
311  /* verify buffer full, set overflow */
312  if(em->bufFull){
313  em->bufFull = 2;
314  }
315  else{
316  uint8_t bufCopy[8];
317 
318  /* prepare data for emergency message */
319  CO_memcpySwap2(&bufCopy[0], &errorCode);
320  bufCopy[2] = 0; /* error register will be set later */
321  bufCopy[3] = errorBit;
322  CO_memcpySwap4(&bufCopy[4], &infoCode);
323 
324  /* copy data to the buffer, increment writePtr and verify buffer full */
325  CO_LOCK_EMCY();
326  CO_memcpy(em->bufWritePtr, &bufCopy[0], 8);
327  em->bufWritePtr += 8;
328 
329  if(em->bufWritePtr == em->bufEnd) em->bufWritePtr = em->buf;
330  if(em->bufWritePtr == em->bufReadPtr) em->bufFull = 1;
331  CO_UNLOCK_EMCY();
332 
333  /* Optional signal to RTOS, which can resume task, which handles CO_EM_process */
334  if(em->pFunctSignal != NULL) {
335  em->pFunctSignal();
336  }
337  }
338  }
339 }
