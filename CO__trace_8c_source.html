111 /* Find variable in Object Dictionary *****************************************/
112 static void findVariable(CO_trace_t *trace) {
113  bool_t err = false;
114  uint16_t index;
115  uint8_t subIndex;
116  uint8_t dataLen;
117  void *OdDataPtr = NULL;
118  int dtIndex = 0;
119 
120  /* parse mapping */
121  index = (uint16_t) ((*trace->map) >> 16);
122  subIndex = (uint8_t) ((*trace->map) >> 8);
123  dataLen = (uint8_t) (*trace->map);
124  if((dataLen & 0x07) != 0) { /* data length must be byte aligned */
125  err = true;
126  }
127  dataLen >>= 3; /* in bytes now */
128  if(dataLen == 0) {
129  dataLen = 4;
130  }
131 
132  /* find mapped variable, if map available */
133  if(!err && (index != 0 || subIndex != 0)) {
134  uint16_t entryNo = CO_OD_find(trace->SDO, index);
135 
136  if(index >= 0x1000 && entryNo != 0xFFFF && subIndex <= trace->SDO->OD[entryNo].maxSubIndex) {
137  OdDataPtr = CO_OD_getDataPointer(trace->SDO, entryNo, subIndex);
138  }
139 
140  if(OdDataPtr != NULL) {
141  uint16_t len = CO_OD_getLength(trace->SDO, entryNo, subIndex);
142 
143  if(len < dataLen) {
144  dataLen = len;
145  }
146  }
147  else {
148  err = true;
149  }
150  }
151 
152  /* Get function pointers for correct data type */
153  if(!err) {
154  /* first sequence: data length */
155  switch(dataLen) {
156  case 1: dtIndex = 0; break;
157  case 2: dtIndex = 1; break;
158  case 4: dtIndex = 2; break;
159  default: err = true; break;
160  }
161  /* second sequence: signed or unsigned */
162  if(((*trace->format) & 1) == 1) {
163  dtIndex += 3;
164  }
165  /* third sequence: Output type */
166  dtIndex += ((*trace->format) >> 1) * 6;
167 
168  if(dtIndex > (sizeof(dataTypes) / sizeof(CO_trace_dataType_t))) {
169  err = true;
170  }
171  }
172 
173  /* set output variables */
174  if(!err) {
175  if(OdDataPtr != NULL) {
176  trace->OD_variable = OdDataPtr;
177  }
178  else {
179  trace->OD_variable = trace->value;
180  }
181  trace->dt = &dataTypes[dtIndex];
182  }
183  else {
184  trace->OD_variable = NULL;
185  trace->dt = NULL;
186  }
187 }

191  * For more information see file CO_SDO.h. */
192 static CO_SDO_abortCode_t CO_ODF_traceConfig(CO_ODF_arg_t *ODF_arg) {
193  CO_trace_t *trace;
194  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
195 
196  trace = (CO_trace_t*) ODF_arg->object;
197 
198  switch(ODF_arg->subIndex) {
199  case 1: /* size */
200  if(ODF_arg->reading) {
201  uint32_t *value = (uint32_t*) ODF_arg->data;
202  *value = trace->bufferSize;
203  }
204  break;
205 
206  case 2: /* axisNo (trace enabled if nonzero) */
207  if(ODF_arg->reading) {
208  uint8_t *value = (uint8_t*) ODF_arg->data;
209  if(!trace->enabled) {
210  *value = 0;
211  }
212  }
213  else {
214  uint8_t *value = (uint8_t*) ODF_arg->data;
215 
216  if(*value == 0) {
217  trace->enabled = false;
218  }
219  else if(!trace->enabled) {
220  if(trace->bufferSize == 0) {
221  ret = CO_SDO_AB_OUT_OF_MEM;
222  }
223  else {
224  /* set trace->OD_variable and trace->dt, based on 'map' and 'format' */
225  findVariable(trace);
226 
227  if(trace->OD_variable != NULL) {
228  *trace->value = 0;
229  *trace->minValue = 0;
230  *trace->maxValue = 0;
231  *trace->triggerTime = 0;
232  trace->valuePrev = 0;
233  trace->readPtr = 0;
234  trace->writePtr = 0;
235  trace->enabled = true;
236  }
237  else {
238  ret = CO_SDO_AB_NO_MAP;
239  }
240  }
241  }
242  }
243  break;
244 
245  case 5: /* map */
246  case 6: /* format */
247  if(!ODF_arg->reading) {
248  if(trace->enabled) {
249  ret = CO_SDO_AB_INVALID_VALUE;
250  }
251  }
252  break;
253  }
254 
255  return ret;
256 }

259 /* OD function for accessing _OD_trace_ (index 0x2400+) from SDO server.
260  * For more information see file CO_SDO.h. */
261 static CO_SDO_abortCode_t CO_ODF_trace(CO_ODF_arg_t *ODF_arg) {
262  CO_trace_t *trace;
263  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
264 
265  trace = (CO_trace_t*) ODF_arg->object;
266 
267  switch(ODF_arg->subIndex) {
268  case 1: /* size */
269  if(ODF_arg->reading) {
270  uint32_t *value = (uint32_t*) ODF_arg->data;
271  uint32_t size = trace->bufferSize;
272  uint32_t wp = trace->writePtr;
273  uint32_t rp = trace->readPtr;
274 
275  if(wp >= rp) {
276  *value = wp - rp;
277  }
278  else {
279  *value = size - rp + wp;
280  }
281  }
282  else {
283  uint32_t *value = (uint32_t*) ODF_arg->data;
284 
285  if(*value == 0) {
286  /* clear buffer, handle race conditions */
287  while(trace->readPtr != 0 || trace->writePtr != 0) {
288  trace->readPtr = 0;
289  trace->writePtr = 0;
290  *trace->triggerTime = 0;
291  }
292  }
293  else {
294  ret = CO_SDO_AB_INVALID_VALUE;
295  }
296  }
297  break;
298 
299  case 5: /* plot */
300  if(ODF_arg->reading) {
301  /* This plot will be transmitted as domain data type. String data
302  * will be printed directly to SDO buffer. If there is more data
303  * to print, than is the size of SDO buffer, then this function
304  * will be called multiple times until internal trace buffer is
305  * empty. Internal trace buffer is circular buffer. It is accessed
306  * by this function and by higher priority thread. If this buffer
307  * is full, there is a danger for race condition. First records
308  * from trace buffer may be overwritten somewhere between. If this
309  * is detected, then do{}while() loop tries printing again. */
310  if(trace->bufferSize == 0 || ODF_arg->dataLength < 100) {
311  ret = CO_SDO_AB_OUT_OF_MEM;
312  }
313  else if(trace->readPtr == trace->writePtr) {
314  ret = CO_SDO_AB_NO_DATA;
315  }
316  else {
317  uint32_t rp, t, v, len, freeLen;
318  char *s;
319  bool_t readPtrOverflowed; /* for handling race conditions */
320 
321  /* repeat everything, if trace->readPtr was overflowed in CO_trace_process */
322  do {
323  readPtrOverflowed = false;
324  s = (char*) ODF_arg->data;
325  freeLen = ODF_arg->dataLength;
326 
327  rp = trace->readPtr;
328 
329  /* start plot, increment variables, verify overflow */
330  if(ODF_arg->firstSegment) {
331  t = trace->timeBuffer[rp];
332  v = trace->valueBuffer[rp];
333  rp ++;
334  if(++trace->readPtr == trace->bufferSize) {
335  trace->readPtr = 0;
336  if(rp != trace->bufferSize) {
337  readPtrOverflowed = true;
338  continue;
339  }
340  rp = 0;
341  }
342  if(rp != trace->readPtr) {
343  readPtrOverflowed = true;
344  continue;
345  }
346  len = trace->dt->printPointStart(s, freeLen, t, v);
347  s += len;
348  freeLen -= len;
349  }
350 
351  /* print other points */
352  if(rp != trace->writePtr) {
353  for(;;) {
354  t = trace->timeBuffer[rp];
355  v = trace->valueBuffer[rp];
356  rp ++;
357  if(++trace->readPtr == trace->bufferSize) {
358  trace->readPtr = 0;
359  if(rp != trace->bufferSize && ODF_arg->firstSegment) {
360  readPtrOverflowed = true;
361  break;
362  }
363  rp = 0;
364  }
365  if(rp != trace->readPtr && ODF_arg->firstSegment) {
366  readPtrOverflowed = true;
367  break;
368  }
369 
370  /* If internal buffer is empty, end transfer */
371  if(rp == trace->writePtr) {
372  /* If there is last time stamp, point will be printed at the end */
373  if(t != trace->lastTimeStamp) {
374  len = trace->dt->printPoint(s, freeLen, t, v);
375  s += len;
376  freeLen -= len;
377  }
378  ODF_arg->lastSegment = true;
379  break;
380  }
381  len = trace->dt->printPoint(s, freeLen, t, v);
382  s += len;
383  freeLen -= len;
384 
385  /* if output buffer is full, next data will be sent later */
386  if(freeLen < 50) {
387  ODF_arg->lastSegment = false;
388  break;
389  }
390  }
391  }
392 
393  /* print last point */
394  if(!readPtrOverflowed && ODF_arg->lastSegment) {
395  v = trace->valuePrev;
396  t = trace->lastTimeStamp;
397  len = trace->dt->printPointEnd(s, freeLen, t, v);
398  s += len;
399  freeLen -= len;
400  }
401  } while(readPtrOverflowed);
402 
403  ODF_arg->dataLength -= freeLen;
404  }
405  }
406  break;
407  }
408 
409  return ret;
410 }

470 /******************************************************************************/
471 void CO_trace_process(CO_trace_t *trace, uint32_t timestamp) {
472  if(trace->enabled) {
473 
474  int32_t val = trace->dt->pGetValue(trace->OD_variable);
475 
476  if(val != trace->valuePrev) {
477  /* Verify, if value passed threshold */
478  if((*trace->trigger & 1) != 0 && trace->valuePrev < *trace->threshold && val >= *trace->threshold) {
479  *trace->triggerTime = timestamp;
480  }
481  if((*trace->trigger & 2) != 0 && trace->valuePrev < *trace->threshold && val >= *trace->threshold) {
482  *trace->triggerTime = timestamp;
483  }
484 
485  /* Write value and verify min/max */
486  if(trace->value != trace->OD_variable) {
487  *trace->value = val;
488  }
489  trace->valuePrev = val;
490  if(*trace->minValue > val) {
491  *trace->minValue = val;
492  }
493  if(*trace->maxValue < val) {
494  *trace->maxValue = val;
495  }
496 
497  /* write buffers and update pointers */
498  trace->timeBuffer[trace->writePtr] = timestamp;
499  trace->valueBuffer[trace->writePtr] = val;
500  if(++trace->writePtr == trace->bufferSize) {
501  trace->writePtr = 0;
502  }
503  if(trace->writePtr == trace->readPtr) {
504  if(++trace->readPtr == trace->bufferSize) {
505  trace->readPtr = 0;
506  }
507  }
508  }
509  else {
510  /* if buffer is empty, make first record */
511  if(trace->writePtr == trace->readPtr) {
512  /* write buffers and update pointers */
513  trace->timeBuffer[trace->writePtr] = timestamp;
514  trace->valueBuffer[trace->writePtr] = val;
515  if(++trace->writePtr == trace->bufferSize) {
516  trace->writePtr = 0;
517  }
518  }
519  }
520  trace->lastTimeStamp = timestamp;
521  }
