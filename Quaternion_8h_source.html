506 template <class Derived>
507 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE typename QuaternionBase<Derived>::Vector3
508 QuaternionBase<Derived>::_transformVector(const Vector3& v) const
509 {
510  // Note that this algorithm comes from the optimization by hand
511  // of the conversion to a Matrix followed by a Matrix/Vector product.
512  // It appears to be much faster than the common algorithm found
513  // in the literature (30 versus 39 flops). It also requires two
514  // Vector3 as temporaries.
515  Vector3 uv = this->vec().cross(v);
516  uv += uv;
517  return v + this->w() * uv + this->vec().cross(uv);
518 }

567 template<class Derived>
568 EIGEN_DEVICE_FUNC inline typename QuaternionBase<Derived>::Matrix3
569 QuaternionBase<Derived>::toRotationMatrix(void) const
570 {
571  // NOTE if inlined, then gcc 4.2 and 4.4 get rid of the temporary (not gcc 4.3 !!)
572  // if not inlined then the cost of the return by value is huge ~ +35%,
573  // however, not inlining this function is an order of magnitude slower, so
574  // it has to be inlined, and so the return by value is not an issue
575  Matrix3 res;
576 
577  const Scalar tx = Scalar(2)*this->x();
578  const Scalar ty = Scalar(2)*this->y();
579  const Scalar tz = Scalar(2)*this->z();
580  const Scalar twx = tx*this->w();
581  const Scalar twy = ty*this->w();
582  const Scalar twz = tz*this->w();
583  const Scalar txx = tx*this->x();
584  const Scalar txy = ty*this->x();
585  const Scalar txz = tz*this->x();
586  const Scalar tyy = ty*this->y();
587  const Scalar tyz = tz*this->y();
588  const Scalar tzz = tz*this->z();
589 
590  res.coeffRef(0,0) = Scalar(1)-(tyy+tzz);
591  res.coeffRef(0,1) = txy-twz;
592  res.coeffRef(0,2) = txz+twy;
593  res.coeffRef(1,0) = txy+twz;
594  res.coeffRef(1,1) = Scalar(1)-(txx+tzz);
595  res.coeffRef(1,2) = tyz-twx;
596  res.coeffRef(2,0) = txz-twy;
597  res.coeffRef(2,1) = tyz+twx;
598  res.coeffRef(2,2) = Scalar(1)-(txx+tyy);
599 
600  return res;
601 }

613 template<class Derived>
614 template<typename Derived1, typename Derived2>
615 EIGEN_DEVICE_FUNC inline Derived& QuaternionBase<Derived>::setFromTwoVectors(const MatrixBase<Derived1>& a, const MatrixBase<Derived2>& b)
616 {
617  EIGEN_USING_STD_MATH(sqrt)
618  Vector3 v0 = a.normalized();
619  Vector3 v1 = b.normalized();
620  Scalar c = v1.dot(v0);
621 
622  // if dot == -1, vectors are nearly opposites
623  // => accurately compute the rotation axis by computing the
624  // intersection of the two planes. This is done by solving:
625  // x^T v0 = 0
626  // x^T v1 = 0
627  // under the constraint:
628  // ||x|| = 1
629  // which yields a singular value problem
630  if (c < Scalar(-1)+NumTraits<Scalar>::dummy_precision())
631  {
632  c = numext::maxi(c,Scalar(-1));
633  Matrix<Scalar,2,3> m; m << v0.transpose(), v1.transpose();
634  JacobiSVD<Matrix<Scalar,2,3> > svd(m, ComputeFullV);
635  Vector3 axis = svd.matrixV().col(2);
636 
637  Scalar w2 = (Scalar(1)+c)*Scalar(0.5);
638  this->w() = sqrt(w2);
639  this->vec() = axis * sqrt(Scalar(1) - w2);
640  return derived();
641  }
642  Vector3 axis = v0.cross(v1);
643  Scalar s = sqrt((Scalar(1)+c)*Scalar(2));
644  Scalar invs = Scalar(1)/s;
645  this->vec() = axis * invs;
646  this->w() = s * Scalar(0.5);
647 
648  return derived();
649 }

762  EIGEN_USING_STD_MATH(sin)
763  const Scalar one = Scalar(1) - NumTraits<Scalar>::epsilon();
764  Scalar d = this->dot(other);
765  Scalar absD = numext::abs(d);
766 
767  Scalar scale0;
768  Scalar scale1;
769 
770  if(absD>=one)
771  {
772  scale0 = Scalar(1) - t;
773  scale1 = t;
774  }
775  else
776  {
777  // theta is the angle between the 2 quaternions
778  Scalar theta = acos(absD);
779  Scalar sinTheta = sin(theta);
780 
781  scale0 = sin( ( Scalar(1) - t ) * theta) / sinTheta;
782  scale1 = sin( ( t * theta) ) / sinTheta;
783  }
784  if(d<Scalar(0)) scale1 = -scale1;
785 
786  return Quaternion<Scalar>(scale0 * coeffs() + scale1 * other.coeffs());
787 }

791 // set from a rotation matrix
792 template<typename Other>
793 struct quaternionbase_assign_impl<Other,3,3>
794 {
795  typedef typename Other::Scalar Scalar;
796  template<class Derived> EIGEN_DEVICE_FUNC static inline void run(QuaternionBase<Derived>& q, const Other& a_mat)
797  {
798  const typename internal::nested_eval<Other,2>::type mat(a_mat);
799  EIGEN_USING_STD_MATH(sqrt)
800  // This algorithm comes from "Quaternion Calculus and Fast Animation",
801  // Ken Shoemake, 1987 SIGGRAPH course notes
802  Scalar t = mat.trace();
803  if (t > Scalar(0))
804  {
805  t = sqrt(t + Scalar(1.0));
806  q.w() = Scalar(0.5)*t;
807  t = Scalar(0.5)/t;
808  q.x() = (mat.coeff(2,1) - mat.coeff(1,2)) * t;
809  q.y() = (mat.coeff(0,2) - mat.coeff(2,0)) * t;
810  q.z() = (mat.coeff(1,0) - mat.coeff(0,1)) * t;
811  }
812  else
813  {
814  Index i = 0;
815  if (mat.coeff(1,1) > mat.coeff(0,0))
816  i = 1;
817  if (mat.coeff(2,2) > mat.coeff(i,i))
818  i = 2;
819  Index j = (i+1)%3;
820  Index k = (j+1)%3;
821 
822  t = sqrt(mat.coeff(i,i)-mat.coeff(j,j)-mat.coeff(k,k) + Scalar(1.0));
823  q.coeffs().coeffRef(i) = Scalar(0.5) * t;
824  t = Scalar(0.5)/t;
825  q.w() = (mat.coeff(k,j)-mat.coeff(j,k))*t;
826  q.coeffs().coeffRef(j) = (mat.coeff(j,i)+mat.coeff(i,j))*t;
827  q.coeffs().coeffRef(k) = (mat.coeff(k,i)+mat.coeff(i,k))*t;
828  }
829  }
830 };
