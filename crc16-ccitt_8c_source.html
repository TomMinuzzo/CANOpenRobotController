105 unsigned short crc16_ccitt(
106  const unsigned char block[],
107  unsigned int blockLength,
108  unsigned short crc)
109 {
110  unsigned int i;
111 
112  for(i=0U; i<blockLength; i++){
113  unsigned short tmp = (crc >> 8) ^ (unsigned short) block[i];
114  crc = ((unsigned short)(crc << 8U)) ^ crc16_ccitt_table[tmp];
115  }
116  return crc;
117 }
