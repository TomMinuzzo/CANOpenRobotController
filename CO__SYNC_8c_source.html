53 /*
54  * Read received message from CAN module.
55  *
56  * Function will be called (by CAN receive interrupt) every time, when CAN
57  * message with correct identifier will be received. For more information and
58  * description of parameters see file CO_driver.h.
59  */
60 static void CO_SYNC_receive(void *object, const CO_CANrxMsg_t *msg){
61  CO_SYNC_t *SYNC;
62  uint8_t operState;
63 
64  SYNC = (CO_SYNC_t*)object; /* this is the correct pointer type of the first argument */
65  operState = *SYNC->operatingState;
66 
67  if((operState == CO_NMT_OPERATIONAL) || (operState == CO_NMT_PRE_OPERATIONAL)){
68  if(SYNC->counterOverflowValue == 0){
69  if(msg->DLC == 0U){
70  SYNC->CANrxNew = true;
71  }
72  else{
73  SYNC->receiveError = (uint16_t)msg->DLC | 0x0100U;
74  }
75  }
76  else{
77  if(msg->DLC == 1U){
78  SYNC->counter = msg->data[0];
79  SYNC->CANrxNew = true;
80  }
81  else{
82  SYNC->receiveError = (uint16_t)msg->DLC | 0x0200U;
83  }
84  }
85  if(SYNC->CANrxNew) {
86  SYNC->CANrxToggle = SYNC->CANrxToggle ? false : true;
87  }
88  }
89 }

92 /*
93  * Function for accessing _COB ID SYNC Message_ (index 0x1005) from SDO server.
94  *
95  * For more information see file CO_SDO.h.
96  */
97 static CO_SDO_abortCode_t CO_ODF_1005(CO_ODF_arg_t *ODF_arg){
98  CO_SYNC_t *SYNC;
99  uint32_t value;
100  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
101 
102  SYNC = (CO_SYNC_t*) ODF_arg->object;
103  value = CO_getUint32(ODF_arg->data);
104 
105  if(!ODF_arg->reading){
106  uint8_t configureSyncProducer = 0;
107 
108  /* only 11-bit CAN identifier is supported */
109  if(value & 0x20000000UL){
110  ret = CO_SDO_AB_INVALID_VALUE;
111  }
112  else{
113  /* is 'generate Sync messge' bit set? */
114  if(value & 0x40000000UL){
115  /* if bit was set before, value can not be changed */
116  if(SYNC->isProducer){
117  ret = CO_SDO_AB_DATA_DEV_STATE;
118  }
119  else{
120  configureSyncProducer = 1;
121  }
122  }
123  }
124 
125  /* configure sync producer and consumer */
126  if(ret == CO_SDO_AB_NONE){
127  SYNC->COB_ID = (uint16_t)(value & 0x7FFU);
128 
129  if(configureSyncProducer){
130  uint8_t len = 0U;
131  if(SYNC->counterOverflowValue != 0U){
132  len = 1U;
133  SYNC->counter = 0U;
134  SYNC->timer = 0U;
135  }
136  SYNC->CANtxBuff = CO_CANtxBufferInit(
137  SYNC->CANdevTx, /* CAN device */
138  SYNC->CANdevTxIdx, /* index of specific buffer inside CAN module */
139  SYNC->COB_ID, /* CAN identifier */
140  0, /* rtr */
141  len, /* number of data bytes */
142  0); /* synchronous message flag bit */
143  SYNC->isProducer = true;
144  }
145  else{
146  SYNC->isProducer = false;
147  }
148 
149  CO_CANrxBufferInit(
150  SYNC->CANdevRx, /* CAN device */
151  SYNC->CANdevRxIdx, /* rx buffer index */
152  SYNC->COB_ID, /* CAN identifier */
153  0x7FF, /* mask */
154  0, /* rtr */
155  (void*)SYNC, /* object passed to receive function */
156  CO_SYNC_receive); /* this function will process received message */
157  }
158  }
159 
160  return ret;
161 }

164 /*
165  * Function for accessing _Communication cycle period_ (index 0x1006) from SDO server.
166  *
167  * For more information see file CO_SDO.h.
168  */
169 static CO_SDO_abortCode_t CO_ODF_1006(CO_ODF_arg_t *ODF_arg){
170  CO_SYNC_t *SYNC;
171  uint32_t value;
172  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
173 
174  SYNC = (CO_SYNC_t*) ODF_arg->object;
175  value = CO_getUint32(ODF_arg->data);
176 
177  if(!ODF_arg->reading){
178  /* period transition from 0 to something */
179  if((SYNC->periodTime == 0) && (value != 0)){
180  SYNC->counter = 0;
181  }
182 
183  SYNC->periodTime = value;
184  SYNC->periodTimeoutTime = (value / 2U) * 3U;
185  /* overflow? */
186  if(SYNC->periodTimeoutTime < value){
187  SYNC->periodTimeoutTime = 0xFFFFFFFFUL;
188  }
189 
190  SYNC->timer = 0;
191  }
192 
193  return ret;
194 }

317 /******************************************************************************/
318 uint8_t CO_SYNC_process(
319  CO_SYNC_t *SYNC,
320  uint32_t timeDifference_us,
321  uint32_t ObjDict_synchronousWindowLength)
322 {
323  uint8_t ret = 0;
324  uint32_t timerNew;
325 
326  if(*SYNC->operatingState == CO_NMT_OPERATIONAL || *SYNC->operatingState == CO_NMT_PRE_OPERATIONAL){
327  /* update sync timer, no overflow */
328  timerNew = SYNC->timer + timeDifference_us;
329  if(timerNew > SYNC->timer) SYNC->timer = timerNew;
330 
331  /* was SYNC just received */
332  if(SYNC->CANrxNew){
333  SYNC->timer = 0;
334  ret = 1;
335  SYNC->CANrxNew = false;
336  }
337 
338  /* SYNC producer */
339  if(SYNC->isProducer && SYNC->periodTime){
340  if(SYNC->timer >= SYNC->periodTime){
341  if(++SYNC->counter > SYNC->counterOverflowValue) SYNC->counter = 1;
342  SYNC->timer = 0;
343  ret = 1;
344  SYNC->CANrxToggle = SYNC->CANrxToggle ? false : true;
345  SYNC->CANtxBuff->data[0] = SYNC->counter;
346  CO_CANsend(SYNC->CANdevTx, SYNC->CANtxBuff);
347  }
348  }
349 
350  /* Synchronous PDOs are allowed only inside time window */
351  if(ObjDict_synchronousWindowLength){
352  if(SYNC->timer > ObjDict_synchronousWindowLength){
353  if(SYNC->curentSyncTimeIsInsideWindow){
354  ret = 2;
355  }
356  SYNC->curentSyncTimeIsInsideWindow = false;
357  }
358  else{
359  SYNC->curentSyncTimeIsInsideWindow = true;
360  }
361  }
362  else{
363  SYNC->curentSyncTimeIsInsideWindow = true;
364  }
365 
366  /* Verify timeout of SYNC */
367  if(SYNC->periodTime && SYNC->timer > SYNC->periodTimeoutTime && *SYNC->operatingState == CO_NMT_OPERATIONAL)
368  CO_errorReport(SYNC->em, CO_EM_SYNC_TIME_OUT, CO_EMC_COMMUNICATION, SYNC->timer);
369  }
370  else {
371  SYNC->CANrxNew = false;
372  }
373 
374  /* verify error from receive function */
375  if(SYNC->receiveError != 0U){
376  CO_errorReport(SYNC->em, CO_EM_SYNC_LENGTH, CO_EMC_SYNC_DATA_LENGTH, (uint32_t)SYNC->receiveError);
377  SYNC->receiveError = 0U;
378  }
379 
380  return ret;
381 }
