76  enum {

78  SrcIsRowMajor = SrcFlags&RowMajorBit,
79  StorageOrdersAgree = (int(DstIsRowMajor) == int(SrcIsRowMajor)),

82  && bool(functor_traits<AssignFunc>::PacketAccess),

84  && int(InnerSize)!=Dynamic && int(InnerSize)%int(InnerPacketSize)==0

88  MayLinearVectorize = bool(MightVectorize) && bool(MayLinearize) && bool(DstHasDirectAccess)
89  && (EIGEN_UNALIGNED_VECTORIZE || (int(DstAlignment)>=int(LinearRequiredAlignment)) || MaxSizeAtCompileTime == Dynamic),

98  };

519 template<typename Kernel>
520 struct dense_assignment_loop<Kernel, SliceVectorizedTraversal, NoUnrolling>
521 {
522  EIGEN_DEVICE_FUNC static EIGEN_STRONG_INLINE void run(Kernel &kernel)
523  {
524  typedef typename Kernel::Scalar Scalar;
525  typedef typename Kernel::PacketType PacketType;
526  enum {
527  packetSize = unpacket_traits<PacketType>::size,
528  requestedAlignment = int(Kernel::AssignmentTraits::InnerRequiredAlignment),
529  alignable = packet_traits<Scalar>::AlignedOnScalar || int(Kernel::AssignmentTraits::DstAlignment)>=sizeof(Scalar),
530  dstIsAligned = int(Kernel::AssignmentTraits::DstAlignment)>=int(requestedAlignment),
531  dstAlignment = alignable ? int(requestedAlignment)
532  : int(Kernel::AssignmentTraits::DstAlignment)
533  };
534  const Scalar *dst_ptr = kernel.dstDataPtr();
535  if((!bool(dstIsAligned)) && (UIntPtr(dst_ptr) % sizeof(Scalar))>0)
536  {
537  // the pointer is not aligned-on scalar, so alignment is not possible
538  return dense_assignment_loop<Kernel,DefaultTraversal,NoUnrolling>::run(kernel);
539  }
540  const Index packetAlignedMask = packetSize - 1;
541  const Index innerSize = kernel.innerSize();
542  const Index outerSize = kernel.outerSize();
543  const Index alignedStep = alignable ? (packetSize - kernel.outerStride() % packetSize) & packetAlignedMask : 0;
544  Index alignedStart = ((!alignable) || bool(dstIsAligned)) ? 0 : internal::first_aligned<requestedAlignment>(dst_ptr, innerSize);
545 
546  for(Index outer = 0; outer < outerSize; ++outer)
547  {
548  const Index alignedEnd = alignedStart + ((innerSize-alignedStart) & ~packetAlignedMask);
549  // do the non-vectorizable part of the assignment
550  for(Index inner = 0; inner<alignedStart ; ++inner)
551  kernel.assignCoeffByOuterInner(outer, inner);
552 
553  // do the vectorizable part of the assignment
554  for(Index inner = alignedStart; inner<alignedEnd; inner+=packetSize)
555  kernel.template assignPacketByOuterInner<dstAlignment, Unaligned, PacketType>(outer, inner);
556 
557  // do the non-vectorizable part of the assignment
558  for(Index inner = alignedEnd; inner<innerSize ; ++inner)
559  kernel.assignCoeffByOuterInner(outer, inner);
560 
561  alignedStart = numext::mini((alignedStart+alignedStep)%packetSize, innerSize);
562  }
563  }
564 };
