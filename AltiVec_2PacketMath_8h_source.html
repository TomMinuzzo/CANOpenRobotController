928 template<typename Packet> EIGEN_STRONG_INLINE void pstoreu_common(__UNPACK_TYPE__(Packet)* to, const Packet& from)
929 {
930  EIGEN_DEBUG_UNALIGNED_STORE
931 #ifdef _BIG_ENDIAN
932  // Taken from http://developer.apple.com/hardwaredrivers/ve/alignment.html
933  // Warning: not thread safe!
934  Packet16uc MSQ, LSQ, edges;
935  Packet16uc edgeAlign, align;
936 
937  MSQ = vec_ld(0, (unsigned char *)to); // most significant quadword
938  LSQ = vec_ld(15, (unsigned char *)to); // least significant quadword
939  edgeAlign = vec_lvsl(0, to); // permute map to extract edges
940  edges=vec_perm(LSQ,MSQ,edgeAlign); // extract the edges
941  align = vec_lvsr( 0, to ); // permute map to misalign data
942  MSQ = vec_perm(edges,(Packet16uc)from,align); // misalign the data (MSQ)
943  LSQ = vec_perm((Packet16uc)from,edges,align); // misalign the data (LSQ)
944  vec_st( LSQ, 15, (unsigned char *)to ); // Store the LSQ part first
945 #else
946  vec_xst(from, 0, to);
947 #endif
948 }
