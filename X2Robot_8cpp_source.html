218  float homingSpeed, float maxTime) {
219  std::vector<bool> success(X2_NUM_JOINTS, false);
220  std::chrono::steady_clock::time_point time0;
221  this->initVelocityControl();
222 
223  for (int i = 0; i < X2_NUM_JOINTS; i++) {
224  if (homingDirection[i] == 0) continue; // skip the joint if it is not asked to do homing
225 
226  Eigen::VectorXd desiredVelocity(X2_NUM_JOINTS);
227  std::chrono::steady_clock::time_point firstTimeHighTorque; // time at the first time joint exceed thresholdTorque
228  bool highTorqueReached = false;
229 
230  desiredVelocity[i] = homingSpeed * homingDirection[i] / std::abs(homingDirection[i]); // setting the desired velocity by using the direction
231  time0 = std::chrono::steady_clock::now();
232 
233  DEBUG_OUT("Homing Joint " << i << "...")
234 
235  while (success[i] == false &&
236  std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - time0).count() < maxTime * 1000) {
237  this->updateRobot(); // because this function has its own loops, updateRobot needs to be called
238  this->setVelocity(desiredVelocity);
239  if (std::abs(this->getTorque()[i]) >= thresholdTorque) { // if high torque is reached
240  highTorqueReached = true;
241  firstTimeHighTorque = std::chrono::steady_clock::now();
242  while (std::chrono::duration_cast<std::chrono::milliseconds> // high torque should be measured for delayTime
243  (std::chrono::steady_clock::now() - firstTimeHighTorque)
244  .count() < delayTime * 1000) {
245  this->updateRobot();
246  if (std::abs(this->getTorque()[i]) < thresholdTorque) { // if torque value reach below thresholdTorque, goes back
247  highTorqueReached = false;
248  break;
249  }
250  }
251  }
252  success[i] = highTorqueReached;
253  }
254 
255  if (success[i]) {
256  DEBUG_OUT("Homing Succeeded for Joint " << i << ".")
257  if (i == X2_LEFT_HIP || i == X2_RIGHT_HIP) { // if it is a hip joint
258 
259  // zeroing is done depending on the limits on the homing direction
260  if (homingDirection[i] > 0)
261  ((X2Joint *)this->joints[i])->setPositionOffset(X2JointLimits.hipMax);
262  else
263  ((X2Joint *)this->joints[i])->setPositionOffset(X2JointLimits.hipMin);
264  } else if (i == X2_LEFT_KNEE || i == X2_RIGHT_KNEE) { // if it is a knee joint
265 
266  // zeroing is done depending on the limits on the homing direction
267  if (homingDirection[i] > 0)
268  ((X2Joint *)this->joints[i])->setPositionOffset(X2JointLimits.kneeMax);
269  else
270  ((X2Joint *)this->joints[i])->setPositionOffset(X2JointLimits.kneeMin);
271  }
272 
273  } else {
274  DEBUG_OUT("Homing Failed for Joint " << i << ".")
275  }
276  }
277  // Checking if all commanded joint successfully homed
278  for (int i = 0; i < X2_NUM_JOINTS; i++) {
279  if (homingDirection[i] == 0) continue; // skip the joint if it is not asked to do homing
280  if (success[i] == false) return false;
281  }
