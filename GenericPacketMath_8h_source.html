518 template<typename Packet> EIGEN_DEVICE_FUNC inline bool predux_any(const Packet& a)
519 {
520  // Dirty but generic implementation where "true" is assumed to be non 0 and all the sames.
521  // It is expected that "true" is either:
522  // - Scalar(1)
523  // - bits full of ones (NaN for floats),
524  // - or first bit equals to 1 (1 for ints, smallest denormal for floats).
525  // For all these cases, taking the sum is just fine, and this boils down to a no-op for scalars.
526  typedef typename unpacket_traits<Packet>::type Scalar;
527  return numext::not_equal_strict(predux(a), Scalar(0));
528 }
