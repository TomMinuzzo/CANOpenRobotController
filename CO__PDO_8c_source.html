55 /*
56  * Read received message from CAN module.
57  *
58  * Function will be called (by CAN receive interrupt) every time, when CAN
59  * message with correct identifier will be received. For more information and
60  * description of parameters see file CO_driver.h.
61  * If new message arrives and previous message wasn't processed yet, then
62  * previous message will be lost and overwritten by new message. That's OK with PDOs.
63  */
64 static void CO_PDO_receive(void *object, const CO_CANrxMsg_t *msg){
65  CO_RPDO_t *RPDO;
66 
67  RPDO = (CO_RPDO_t*)object; /* this is the correct pointer type of the first argument */
68 
69  if( (RPDO->valid) &&
70  (*RPDO->operatingState == CO_NMT_OPERATIONAL) &&
71  (msg->DLC >= RPDO->dataLength))
72  {
73  if(RPDO->synchronous && RPDO->SYNC->CANrxToggle) {
74  /* copy data into second buffer and set 'new message' flag */
75  RPDO->CANrxData[1][0] = msg->data[0];
76  RPDO->CANrxData[1][1] = msg->data[1];
77  RPDO->CANrxData[1][2] = msg->data[2];
78  RPDO->CANrxData[1][3] = msg->data[3];
79  RPDO->CANrxData[1][4] = msg->data[4];
80  RPDO->CANrxData[1][5] = msg->data[5];
81  RPDO->CANrxData[1][6] = msg->data[6];
82  RPDO->CANrxData[1][7] = msg->data[7];
83 
84  RPDO->CANrxNew[1] = true;
85  }
86  else {
87  /* copy data into default buffer and set 'new message' flag */
88  RPDO->CANrxData[0][0] = msg->data[0];
89  RPDO->CANrxData[0][1] = msg->data[1];
90  RPDO->CANrxData[0][2] = msg->data[2];
91  RPDO->CANrxData[0][3] = msg->data[3];
92  RPDO->CANrxData[0][4] = msg->data[4];
93  RPDO->CANrxData[0][5] = msg->data[5];
94  RPDO->CANrxData[0][6] = msg->data[6];
95  RPDO->CANrxData[0][7] = msg->data[7];
96 
97  RPDO->CANrxNew[0] = true;
98  }
99  }
100 }

103 /*
104  * Configure RPDO Communication parameter.
105  *
106  * Function is called from commuincation reset or when parameter changes.
107  *
108  * Function configures following variable from CO_RPDO_t: _valid_. It also
109  * configures CAN rx buffer. If configuration fails, emergency message is send
110  * and device is not able to enter NMT operational.
111  *
112  * @param RPDO RPDO object.
113  * @param COB_IDUsedByRPDO _RPDO communication parameter_, _COB-ID for PDO_ variable
114  * from Object dictionary (index 0x1400+, subindex 1).
115  */
116 static void CO_RPDOconfigCom(CO_RPDO_t* RPDO, uint32_t COB_IDUsedByRPDO){
117  uint16_t ID;
118  CO_ReturnError_t r;
119 
120  ID = (uint16_t)COB_IDUsedByRPDO;
121 
122  /* is RPDO used? */
123  if((COB_IDUsedByRPDO & 0xBFFFF800L) == 0 && RPDO->dataLength && ID){
124  /* is used default COB-ID? */
125  if(ID == RPDO->defaultCOB_ID) ID += RPDO->nodeId;
126  RPDO->valid = true;
127  RPDO->synchronous = (RPDO->RPDOCommPar->transmissionType <= 240) ? true : false;
128  }
129  else{
130  ID = 0;
131  RPDO->valid = false;
132  RPDO->CANrxNew[0] = RPDO->CANrxNew[1] = false;
133  }
134  r = CO_CANrxBufferInit(
135  RPDO->CANdevRx, /* CAN device */
136  RPDO->CANdevRxIdx, /* rx buffer index */
137  ID, /* CAN identifier */
138  0x7FF, /* mask */
139  0, /* rtr */
140  (void*)RPDO, /* object passed to receive function */
141  CO_PDO_receive); /* this function will process received message */
142  if(r != CO_ERROR_NO){
143  RPDO->valid = false;
144  RPDO->CANrxNew[0] = RPDO->CANrxNew[1] = false;
145  }
146 }

193 /*
194  * Find mapped variable in Object Dictionary.
195  *
196  * Function is called from CO_R(T)PDOconfigMap or when mapping parameter changes.
197  *
198  * @param SDO SDO object.
199  * @param map PDO mapping parameter.
200  * @param R_T 0 for RPDO map, 1 for TPDO map.
201  * @param ppData Pointer to returning parameter: pointer to data of mapped variable.
202  * @param pLength Pointer to returning parameter: *add* length of mapped variable.
203  * @param pSendIfCOSFlags Pointer to returning parameter: sendIfCOSFlags variable.
204  * @param pIsMultibyteVar Pointer to returning parameter: true for multibyte variable.
205  *
206  * @return 0 on success, otherwise SDO abort code.
207  */
208 static CO_SDO_abortCode_t CO_PDOfindMap(
209  CO_SDO_t *SDO,
210  uint32_t map,
211  uint8_t R_T,
212  uint8_t **ppData,
213  uint8_t *pLength,
214  uint8_t *pSendIfCOSFlags,
215  uint8_t *pIsMultibyteVar)
216 {
217  uint16_t entryNo;
218  uint16_t index;
219  uint8_t subIndex;
220  uint8_t dataLen;
221  uint8_t objectLen;
222  uint8_t attr;
223 
224  index = (uint16_t)(map>>16);
225  subIndex = (uint8_t)(map>>8);
226  dataLen = (uint8_t) map; /* data length in bits */
227 
228  /* data length must be byte aligned */
229  if(dataLen&0x07) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
230 
231  dataLen >>= 3; /* new data length is in bytes */
232  *pLength += dataLen;
233 
234  /* total PDO length can not be more than 8 bytes */
235  if(*pLength > 8) return CO_SDO_AB_MAP_LEN; /* The number and length of the objects to be mapped would exceed PDO length. */
236 
237  /* is there a reference to dummy entries */
238  if(index <=7 && subIndex == 0){
239  static uint32_t dummyTX = 0;
240  static uint32_t dummyRX;
241  uint8_t dummySize = 4;
242 
243  if(index<2) dummySize = 0;
244  else if(index==2 || index==5) dummySize = 1;
245  else if(index==3 || index==6) dummySize = 2;
246 
247  /* is size of variable big enough for map */
248  if(dummySize < dataLen) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
249 
250  /* Data and ODE pointer */
251  if(R_T == 0) *ppData = (uint8_t*) &dummyRX;
252  else *ppData = (uint8_t*) &dummyTX;
253 
254  return CO_SDO_AB_NONE;
255  }
256 
257  /* find object in Object Dictionary */
258  entryNo = CO_OD_find(SDO, index);
259 
260  /* Does object exist in OD? */
261  if(entryNo == 0xFFFF || subIndex > SDO->OD[entryNo].maxSubIndex)
262  return CO_SDO_AB_NOT_EXIST; /* Object does not exist in the object dictionary. */
263 
264  attr = CO_OD_getAttribute(SDO, entryNo, subIndex);
265  /* Is object Mappable for RPDO? */
266  if(R_T==0 && !((attr&CO_ODA_RPDO_MAPABLE) && (attr&CO_ODA_WRITEABLE))) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
267  /* Is object Mappable for TPDO? */
268  if(R_T!=0 && !((attr&CO_ODA_TPDO_MAPABLE) && (attr&CO_ODA_READABLE))) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
269 
270  /* is size of variable big enough for map */
271  objectLen = CO_OD_getLength(SDO, entryNo, subIndex);
272  if(objectLen < dataLen) return CO_SDO_AB_NO_MAP; /* Object cannot be mapped to the PDO. */
273 
274  /* mark multibyte variable */
275  *pIsMultibyteVar = (attr&CO_ODA_MB_VALUE) ? 1 : 0;
276 
277  /* pointer to data */
278  *ppData = (uint8_t*) CO_OD_getDataPointer(SDO, entryNo, subIndex);
279 #ifdef CO_BIG_ENDIAN
280  /* skip unused MSB bytes */
281  if(*pIsMultibyteVar){
282  *ppData += objectLen - dataLen;
283  }
284 #endif
285 
286  /* setup change of state flags */
287  if(attr&CO_ODA_TPDO_DETECT_COS){
288  int16_t i;
289  for(i=*pLength-dataLen; i<*pLength; i++){
290  *pSendIfCOSFlags |= 1<<i;
291  }
292  }
293 
294  return CO_SDO_AB_NONE;
295 }

298 /*
299  * Configure RPDO Mapping parameter.
300  *
301  * Function is called from communication reset or when parameter changes.
302  *
303  * Function configures following variables from CO_RPDO_t: _dataLength_ and
304  * _mapPointer_.
305  *
306  * @param RPDO RPDO object.
307  * @param noOfMappedObjects Number of mapped object (from OD).
308  *
309  * @return 0 on success, otherwise SDO abort code.
310  */
311 static CO_SDO_abortCode_t CO_RPDOconfigMap(CO_RPDO_t* RPDO, uint8_t noOfMappedObjects){
312  int16_t i;
313  uint8_t length = 0;
314  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
315  const uint32_t* pMap = &RPDO->RPDOMapPar->mappedObject1;
316 
317  for(i=noOfMappedObjects; i>0; i--){
318  int16_t j;
319  uint8_t* pData;
320  uint8_t dummy = 0;
321  uint8_t prevLength = length;
322  uint8_t MBvar;
323  uint32_t map = *(pMap++);
324 
325  /* function do much checking of errors in map */
326  ret = CO_PDOfindMap(
327  RPDO->SDO,
328  map,
329  0,
330  &pData,
331  &length,
332  &dummy,
333  &MBvar);
334  if(ret){
335  length = 0;
336  CO_errorReport(RPDO->em, CO_EM_PDO_WRONG_MAPPING, CO_EMC_PROTOCOL_ERROR, map);
337  break;
338  }
339 
340  /* write PDO data pointers */
341 #ifdef CO_BIG_ENDIAN
342  if(MBvar){
343  for(j=length-1; j>=prevLength; j--)
344  RPDO->mapPointer[j] = pData++;
345  }
346  else{
347  for(j=prevLength; j<length; j++)
348  RPDO->mapPointer[j] = pData++;
349  }
350 #else
351  for(j=prevLength; j<length; j++){
352  RPDO->mapPointer[j] = pData++;
353  }
354 #endif
355 
356  }
357 
358  RPDO->dataLength = length;
359 
360  return ret;
361 }

431 /*
432  * Function for accessing _RPDO communication parameter_ (index 0x1400+) from SDO server.
433  *
434  * For more information see file CO_SDO.h.
435  */
436 static CO_SDO_abortCode_t CO_ODF_RPDOcom(CO_ODF_arg_t *ODF_arg){
437  CO_RPDO_t *RPDO;
438 
439  RPDO = (CO_RPDO_t*) ODF_arg->object;
440 
441  /* Reading Object Dictionary variable */
442  if(ODF_arg->reading){
443  if(ODF_arg->subIndex == 1){
444  uint32_t *value = (uint32_t*) ODF_arg->data;
445 
446  /* if default COB ID is used, write default value here */
447  if(((*value)&0xFFFF) == RPDO->defaultCOB_ID && RPDO->defaultCOB_ID)
448  *value += RPDO->nodeId;
449 
450  /* If PDO is not valid, set bit 31 */
451  if(!RPDO->valid) *value |= 0x80000000L;
452  }
453  return CO_SDO_AB_NONE;
454  }
455 
456  /* Writing Object Dictionary variable */
457  if(RPDO->restrictionFlags & 0x04)
458  return CO_SDO_AB_READONLY; /* Attempt to write a read only object. */
459  if(*RPDO->operatingState == CO_NMT_OPERATIONAL && (RPDO->restrictionFlags & 0x01))
460  return CO_SDO_AB_DATA_DEV_STATE; /* Data cannot be transferred or stored to the application because of the present device state. */
461 
462  if(ODF_arg->subIndex == 1){ /* COB_ID */
463  uint32_t *value = (uint32_t*) ODF_arg->data;
464 
465  /* bits 11...29 must be zero */
466  if(*value & 0x3FFF8000L)
467  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
468 
469  /* if default COB-ID is being written, write defaultCOB_ID without nodeId */
470  if(((*value)&0xFFFF) == (RPDO->defaultCOB_ID + RPDO->nodeId)){
471  *value &= 0xC0000000L;
472  *value += RPDO->defaultCOB_ID;
473  }
474 
475  /* if PDO is valid, bits 0..29 can not be changed */
476  if(RPDO->valid && ((*value ^ RPDO->RPDOCommPar->COB_IDUsedByRPDO) & 0x3FFFFFFFL))
477  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
478 
479  /* configure RPDO */
480  CO_RPDOconfigCom(RPDO, *value);
481  }
482  else if(ODF_arg->subIndex == 2){ /* Transmission_type */
483  uint8_t *value = (uint8_t*) ODF_arg->data;
484  bool_t synchronousPrev = RPDO->synchronous;
485 
486  /* values from 241...253 are not valid */
487  if(*value >= 241 && *value <= 253)
488  return CO_SDO_AB_INVALID_VALUE; /* Invalid value for parameter (download only). */
489 
490  RPDO->synchronous = (*value <= 240) ? true : false;
491 
492  /* Remove old message from second buffer. */
493  if(RPDO->synchronous != synchronousPrev) {
494  RPDO->CANrxNew[1] = false;
495  }
496  }
497 
498  return CO_SDO_AB_NONE;
499 }
