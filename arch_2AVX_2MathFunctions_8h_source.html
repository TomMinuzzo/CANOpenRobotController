72 // Functions for sqrt.
73 // The EIGEN_FAST_MATH version uses the _mm_rsqrt_ps approximation and one step
74 // of Newton's method, at a cost of 1-2 bits of precision as opposed to the
75 // exact solution. It does not handle +inf, or denormalized numbers correctly.
76 // The main advantage of this approach is not just speed, but also the fact that
77 // it can be inlined and pipelined with other computations, further reducing its
78 // effective latency. This is similar to Quake3's fast inverse square root.
79 // For detail see here: http://www.beyond3d.com/content/articles/8/
80 #if EIGEN_FAST_MATH
81 template <>
82 EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8f
83 psqrt<Packet8f>(const Packet8f& _x) {
84  Packet8f half = pmul(_x, pset1<Packet8f>(.5f));
85  Packet8f denormal_mask = _mm256_and_ps(
86  _mm256_cmp_ps(_x, pset1<Packet8f>((std::numeric_limits<float>::min)()),
87  _CMP_LT_OQ),
88  _mm256_cmp_ps(_x, _mm256_setzero_ps(), _CMP_GE_OQ));
89 
90  // Compute approximate reciprocal sqrt.
91  Packet8f x = _mm256_rsqrt_ps(_x);
92  // Do a single step of Newton's iteration.
93  x = pmul(x, psub(pset1<Packet8f>(1.5f), pmul(half, pmul(x,x))));
94  // Flush results for denormals to zero.
95  return _mm256_andnot_ps(denormal_mask, pmul(_x,x));
96 }
97 #else

109 template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
110 Packet8f prsqrt<Packet8f>(const Packet8f& _x) {
111  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(inf, 0x7f800000);
112  _EIGEN_DECLARE_CONST_Packet8f(one_point_five, 1.5f);
113  _EIGEN_DECLARE_CONST_Packet8f(minus_half, -0.5f);
114  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(flt_min, 0x00800000);
115 
116  Packet8f neg_half = pmul(_x, p8f_minus_half);
117 
118  // select only the inverse sqrt of positive normal inputs (denormals are
119  // flushed to zero and cause infs as well).
120  Packet8f lt_min_mask = _mm256_cmp_ps(_x, p8f_flt_min, _CMP_LT_OQ);
121  Packet8f inf_mask = _mm256_cmp_ps(_x, p8f_inf, _CMP_EQ_OQ);
122  Packet8f not_normal_finite_mask = _mm256_or_ps(lt_min_mask, inf_mask);
123 
124  // Compute an approximate result using the rsqrt intrinsic.
125  Packet8f y_approx = _mm256_rsqrt_ps(_x);
126 
127  // Do a single step of Newton-Raphson iteration to improve the approximation.
128  // This uses the formula y_{n+1} = y_n * (1.5 - y_n * (0.5 * x) * y_n).
129  // It is essential to evaluate the inner term like this because forming
130  // y_n^2 may over- or underflow.
131  Packet8f y_newton = pmul(y_approx, pmadd(y_approx, pmul(neg_half, y_approx), p8f_one_point_five));
132 
133  // Select the result of the Newton-Raphson step for positive normal arguments.
134  // For other arguments, choose the output of the intrinsic. This will
135  // return rsqrt(+inf) = 0, rsqrt(x) = NaN if x < 0, and rsqrt(x) = +inf if
136  // x is zero or a positive denormalized float (equivalent to flushing positive
137  // denormalized inputs to zero).
138  return pselect<Packet8f>(not_normal_finite_mask, y_approx, y_newton);
139 }
