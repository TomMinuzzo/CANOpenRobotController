103 template<> EIGEN_STRONG_INLINE Packet2cf pmul<Packet2cf>(const Packet2cf& a, const Packet2cf& b)
104 {
105  Packet4f v1, v2;
106 
107  // Permute and multiply the real parts of a and b
108  v1 = vec_perm(a.v, a.v, p16uc_PSET32_WODD);
109  // Get the imaginary parts of a
110  v2 = vec_perm(a.v, a.v, p16uc_PSET32_WEVEN);
111  // multiply a_re * b
112  v1 = vec_madd(v1, b.v, p4f_ZERO);
113  // multiply a_im * b and get the conjugate result
114  v2 = vec_madd(v2, b.v, p4f_ZERO);
115  v2 = reinterpret_cast<Packet4f>(pxor(v2, reinterpret_cast<Packet4f>(p4ui_CONJ_XOR)));
116  // permute back to a proper order
117  v2 = vec_perm(v2, v2, p16uc_COMPLEX32_REV);
118 
119  return Packet2cf(padd<Packet4f>(v1, v2));
120 }

382 template<> EIGEN_STRONG_INLINE Packet1cd pcmp_eq(const Packet1cd& a, const Packet1cd& b) {
383  // Compare real and imaginary parts of a and b to get the mask vector:
384  // [re(a)==re(b), im(a)==im(b)]
385  Packet2d eq = reinterpret_cast<Packet2d>(vec_cmpeq(a.v,b.v));
386  // Swap real/imag elements in the mask in to get:
387  // [im(a)==im(b), re(a)==re(b)]
388  Packet2d eq_swapped = reinterpret_cast<Packet2d>(vec_sld(reinterpret_cast<Packet4ui>(eq), reinterpret_cast<Packet4ui>(eq), 8));
389  // Return re(a)==re(b) & im(a)==im(b) by computing bitwise AND of eq and eq_swapped
390  return Packet1cd(vec_and(eq, eq_swapped));
