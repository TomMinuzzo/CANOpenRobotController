407 EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC __half_raw float_to_half_rtne(float ff) {
408 #if (defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300) || \
409  (defined(EIGEN_HAS_HIP_FP16) && defined(EIGEN_HIP_DEVICE_COMPILE))
410  __half tmp_ff = __float2half(ff);
411  return *(__half_raw*)&tmp_ff;
412 
413 #elif defined(EIGEN_HAS_FP16_C)
414  __half_raw h;
415  h.x = _cvtss_sh(ff, 0);
416  return h;
417 
418 #else
419  float32_bits f; f.f = ff;
420 
421  const float32_bits f32infty = { 255 << 23 };
422  const float32_bits f16max = { (127 + 16) << 23 };
423  const float32_bits denorm_magic = { ((127 - 15) + (23 - 10) + 1) << 23 };
424  unsigned int sign_mask = 0x80000000u;
425  __half_raw o;
426  o.x = static_cast<unsigned short>(0x0u);
427 
428  unsigned int sign = f.u & sign_mask;
429  f.u ^= sign;
430 
431  // NOTE all the integer compares in this function can be safely
432  // compiled into signed compares since all operands are below
433  // 0x80000000. Important if you want fast straight SSE2 code
434  // (since there's no unsigned PCMPGTD).
435 
436  if (f.u >= f16max.u) { // result is Inf or NaN (all exponent bits set)
437  o.x = (f.u > f32infty.u) ? 0x7e00 : 0x7c00; // NaN->qNaN and Inf->Inf
438  } else { // (De)normalized number or zero
439  if (f.u < (113 << 23)) { // resulting FP16 is subnormal or zero
440  // use a magic value to align our 10 mantissa bits at the bottom of
441  // the float. as long as FP addition is round-to-nearest-even this
442  // just works.
443  f.f += denorm_magic.f;
444 
445  // and one integer subtract of the bias later, we have our final float!
446  o.x = static_cast<unsigned short>(f.u - denorm_magic.u);
447  } else {
448  unsigned int mant_odd = (f.u >> 13) & 1; // resulting mantissa is odd
449 
450  // update exponent, rounding bias part 1
451  f.u += ((unsigned int)(15 - 127) << 23) + 0xfff;
452  // rounding bias part 2
453  f.u += mant_odd;
454  // take the bits!
455  o.x = static_cast<unsigned short>(f.u >> 13);
456  }
457  }
458 
459  o.x |= static_cast<unsigned short>(sign >> 16);
460  return o;
461 #endif
462 }

464 EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC float half_to_float(__half_raw h) {
465 #if (defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300) || \
466  (defined(EIGEN_HAS_HIP_FP16) && defined(EIGEN_HIP_DEVICE_COMPILE))
467  return __half2float(h);
468 
469 #elif defined(EIGEN_HAS_FP16_C)
470  return _cvtsh_ss(h.x);
471 
472 #else
473  const float32_bits magic = { 113 << 23 };
474  const unsigned int shifted_exp = 0x7c00 << 13; // exponent mask after shift
475  float32_bits o;
476 
477  o.u = (h.x & 0x7fff) << 13; // exponent/mantissa bits
478  unsigned int exp = shifted_exp & o.u; // just the exponent
479  o.u += (127 - 15) << 23; // exponent adjust
480 
481  // handle exponent special cases
482  if (exp == shifted_exp) { // Inf/NaN?
483  o.u += (128 - 16) << 23; // extra exp adjust
484  } else if (exp == 0) { // Zero/Denormal?
485  o.u += 1 << 23; // extra exp adjust
486  o.f -= magic.f; // renormalize
487  }
488 
489  o.u |= (h.x & 0x8000) << 16; // sign bit
490  return o.f;
491 #endif
492 }
