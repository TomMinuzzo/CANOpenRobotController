78 static void CO_HBcons_monitoredNodeConfig(
79  CO_HBconsumer_t *HBcons,
80  uint8_t idx,
81  uint32_t HBconsTime)
82 {
83  uint16_t COB_ID;
84  uint16_t NodeID;
85  CO_HBconsNode_t *monitoredNode;
86 
87  if(idx >= HBcons->numberOfMonitoredNodes) return;
88 
89  NodeID = (uint16_t)((HBconsTime>>16)&0xFF);
90  monitoredNode = &HBcons->monitoredNodes[idx];
91  monitoredNode->time = (uint16_t)HBconsTime;
92  monitoredNode->NMTstate = 0;
93  monitoredNode->monStarted = false;
94 
95  /* is channel used */
96  if(NodeID && monitoredNode->time){
97  COB_ID = NodeID + 0x700;
98  }
99  else{
100  COB_ID = 0;
101  monitoredNode->time = 0;
102  }
103 
104  /* configure Heartbeat consumer CAN reception */
105  CO_CANrxBufferInit(
106  HBcons->CANdevRx,
107  HBcons->CANdevRxIdxStart + idx,
108  COB_ID,
109  0x7FF,
110  0,
111  (void*)&HBcons->monitoredNodes[idx],
112  CO_HBcons_receive);
113 }

116 /*
117  * OD function for accessing _Consumer Heartbeat Time_ (index 0x1016) from SDO server.
118  *
119  * For more information see file CO_SDO.h.
120  */
121 static CO_SDO_abortCode_t CO_ODF_1016(CO_ODF_arg_t *ODF_arg);
122 static CO_SDO_abortCode_t CO_ODF_1016(CO_ODF_arg_t *ODF_arg){
123  CO_HBconsumer_t *HBcons;
124  uint32_t value;
125  CO_SDO_abortCode_t ret = CO_SDO_AB_NONE;
126 
127  HBcons = (CO_HBconsumer_t*) ODF_arg->object;
128  value = CO_getUint32(ODF_arg->data);
129 
130  if(!ODF_arg->reading){
131  uint8_t NodeID;
132  uint16_t HBconsTime;
133 
134  NodeID = (value >> 16U) & 0xFFU;
135  HBconsTime = value & 0xFFFFU;
136 
137  if((value & 0xFF800000U) != 0){
138  ret = CO_SDO_AB_PRAM_INCOMPAT;
139  }
140  else if((HBconsTime != 0) && (NodeID != 0)){
141  uint8_t i;
142  /* there must not be more entries with same index and time different than zero */
143  for(i = 0U; i<HBcons->numberOfMonitoredNodes; i++){
144  uint32_t objectCopy = HBcons->HBconsTime[i];
145  uint8_t NodeIDObj = (objectCopy >> 16U) & 0xFFU;
146  uint16_t HBconsTimeObj = objectCopy & 0xFFFFU;
147  if(((ODF_arg->subIndex-1U) != i) && (HBconsTimeObj != 0) && (NodeID == NodeIDObj)){
148  ret = CO_SDO_AB_PRAM_INCOMPAT;
149  }
150  }
151  }
152  else{
153  ret = CO_SDO_AB_NONE;
154  }
155 
156  /* Configure */
157  if(ret == CO_SDO_AB_NONE){
158  CO_HBcons_monitoredNodeConfig(HBcons, ODF_arg->subIndex-1U, value);
159  }
160  }
161 
162  return ret;
163 }

204 /******************************************************************************/
205 void CO_HBconsumer_process(
206  CO_HBconsumer_t *HBcons,
207  bool_t NMTisPreOrOperational,
208  uint16_t timeDifference_ms)
209 {
210  uint8_t i;
211  uint8_t AllMonitoredOperationalCopy;
212  CO_HBconsNode_t *monitoredNode;
213 
214  AllMonitoredOperationalCopy = 5;
215  monitoredNode = &HBcons->monitoredNodes[0];
216 
217  if(NMTisPreOrOperational){
218  for(i=0; i<HBcons->numberOfMonitoredNodes; i++){
219  if(monitoredNode->time){/* is node monitored */
220  /* Verify if new Consumer Heartbeat message received */
221  if(monitoredNode->CANrxNew){
222  if(monitoredNode->NMTstate){
223  /* not a bootup message */
224  monitoredNode->monStarted = true;
225  monitoredNode->timeoutTimer = 0; /* reset timer */
226  timeDifference_ms = 0;
227  }
228  monitoredNode->CANrxNew = false;
229  }
230  /* Verify timeout */
231  if(monitoredNode->timeoutTimer < monitoredNode->time) monitoredNode->timeoutTimer += timeDifference_ms;
232 
233  if(monitoredNode->monStarted){
234  if(monitoredNode->timeoutTimer >= monitoredNode->time){
235  CO_errorReport(HBcons->em, CO_EM_HEARTBEAT_CONSUMER, CO_EMC_HEARTBEAT, i);
236  monitoredNode->NMTstate = 0;
237  }
238  else if(monitoredNode->NMTstate == 0){
239  /* there was a bootup message */
240  CO_errorReport(HBcons->em, CO_EM_HB_CONSUMER_REMOTE_RESET, CO_EMC_HEARTBEAT, i);
241  }
242  }
243  if(monitoredNode->NMTstate != CO_NMT_OPERATIONAL)
244  AllMonitoredOperationalCopy = 0;
245  }
246  monitoredNode++;
247  }
248  }
249  else{ /* not in (pre)operational state */
250  for(i=0; i<HBcons->numberOfMonitoredNodes; i++){
251  monitoredNode->NMTstate = 0;
252  monitoredNode->CANrxNew = false;
253  monitoredNode->monStarted = false;
254  monitoredNode++;
255  }
256  AllMonitoredOperationalCopy = 0;
257  }
258  HBcons->allMonitoredOperational = AllMonitoredOperationalCopy;
259 }
