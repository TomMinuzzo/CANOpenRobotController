52 /*
53  * Read received message from CAN module.
54  *
55  * Function will be called (by CAN receive interrupt) every time, when CAN
56  * message with correct identifier will be received. For more information and
57  * description of parameters see file CO_driver.h.
58  */
59 static void CO_NMT_receive(void *object, const CO_CANrxMsg_t *msg){
60  CO_NMT_t *NMT;
61  uint8_t nodeId;
62 
63  NMT = (CO_NMT_t*)object; /* this is the correct pointer type of the first argument */
64 
65  nodeId = msg->data[1];
66 
67  if((msg->DLC == 2) && ((nodeId == 0) || (nodeId == NMT->nodeId))){
68  uint8_t command = msg->data[0];
69  uint8_t currentOperatingState = NMT->operatingState;
70 
71  switch(command){
72  case CO_NMT_ENTER_OPERATIONAL:
73  if((*NMT->emPr->errorRegister) == 0U){
74  NMT->operatingState = CO_NMT_OPERATIONAL;
75  }
76  break;
77  case CO_NMT_ENTER_STOPPED:
78  NMT->operatingState = CO_NMT_STOPPED;
79  break;
80  case CO_NMT_ENTER_PRE_OPERATIONAL:
81  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
82  break;
83  case CO_NMT_RESET_NODE:
84  NMT->resetCommand = CO_RESET_APP;
85  break;
86  case CO_NMT_RESET_COMMUNICATION:
87  NMT->resetCommand = CO_RESET_COMM;
88  break;
89  }
90 
91  if(NMT->pFunctNMT!=NULL && currentOperatingState!=NMT->operatingState){
92  NMT->pFunctNMT(NMT->operatingState);
93  }
94  }
95 }

98 /******************************************************************************/
99 CO_ReturnError_t CO_NMT_init(
100  CO_NMT_t *NMT,
101  CO_EMpr_t *emPr,
102  uint8_t nodeId,
103  uint16_t firstHBTime,
104  CO_CANmodule_t *NMT_CANdev,
105  uint16_t NMT_rxIdx,
106  uint16_t CANidRxNMT,
107  CO_CANmodule_t *HB_CANdev,
108  uint16_t HB_txIdx,
109  uint16_t CANidTxHB)
110 {
111  /* verify arguments */
112  if(NMT==NULL || emPr==NULL || NMT_CANdev==NULL || HB_CANdev==NULL){
113  return CO_ERROR_ILLEGAL_ARGUMENT;
114  }
115 
116  /* blinking bytes */
117  NMT->LEDflickering = 0;
118  NMT->LEDblinking = 0;
119  NMT->LEDsingleFlash = 0;
120  NMT->LEDdoubleFlash = 0;
121  NMT->LEDtripleFlash = 0;
122  NMT->LEDquadrupleFlash = 0;
123 
124  /* Configure object variables */
125  NMT->operatingState = CO_NMT_INITIALIZING;
126  NMT->LEDgreenRun = -1;
127  NMT->LEDredError = 1;
128  NMT->nodeId = nodeId;
129  NMT->firstHBTime = firstHBTime;
130  NMT->resetCommand = CO_RESET_NOT;
131  NMT->HBproducerTimer = 0xFFFF;
132  NMT->emPr = emPr;
133  NMT->pFunctNMT = NULL;
134 
135  /* configure NMT CAN reception */
136  CO_CANrxBufferInit(
137  NMT_CANdev, /* CAN device */
138  NMT_rxIdx, /* rx buffer index */
139  CANidRxNMT, /* CAN identifier */
140  0x7FF, /* mask */
141  0, /* rtr */
142  (void*)NMT, /* object passed to receive function */
143  CO_NMT_receive); /* this function will process received message */
144 
145  /* configure HB CAN transmission */
146  NMT->HB_CANdev = HB_CANdev;
147  NMT->HB_TXbuff = CO_CANtxBufferInit(
148  HB_CANdev, /* CAN device */
149  HB_txIdx, /* index of specific buffer inside CAN module */
150  CANidTxHB, /* CAN identifier */
151  0, /* rtr */
152  1, /* number of data bytes */
153  0); /* synchronous message flag bit */
154 
155  return CO_ERROR_NO;
156 }

208 /******************************************************************************/
209 CO_NMT_reset_cmd_t CO_NMT_process(
210  CO_NMT_t *NMT,
211  uint16_t timeDifference_ms,
212  uint16_t HBtime,
213  uint32_t NMTstartup,
214  uint8_t errorRegister,
215  const uint8_t errorBehavior[],
216  uint16_t *timerNext_ms)
217 {
218  uint8_t CANpassive;
219 
220  uint8_t currentOperatingState = NMT->operatingState;
221 
222  NMT->HBproducerTimer += timeDifference_ms;
223 
224  /* Heartbeat producer message & Bootup message */
225  if((HBtime != 0 && NMT->HBproducerTimer >= HBtime) || NMT->operatingState == CO_NMT_INITIALIZING){
226 
227  /* Start from the beginning. If OS is slow, time sliding may occur. However, heartbeat is
228  * not for synchronization, it is for health report. */
229  NMT->HBproducerTimer = 0;
230 
231  NMT->HB_TXbuff->data[0] = NMT->operatingState;
232  CO_CANsend(NMT->HB_CANdev, NMT->HB_TXbuff);
233 
234  if(NMT->operatingState == CO_NMT_INITIALIZING){
235  if(HBtime > NMT->firstHBTime) NMT->HBproducerTimer = HBtime - NMT->firstHBTime;
236  else NMT->HBproducerTimer = 0;
237 
238  if((NMTstartup & 0x04) == 0) NMT->operatingState = CO_NMT_OPERATIONAL;
239  else NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
240  }
241  }
242 
243 
244  /* Calculate, when next Heartbeat needs to be send and lower timerNext_ms if necessary. */
245  if(HBtime != 0 && timerNext_ms != NULL){
246  if(NMT->HBproducerTimer < HBtime){
247  uint16_t diff = HBtime - NMT->HBproducerTimer;
248  if(*timerNext_ms > diff){
249  *timerNext_ms = diff;
250  }
251  }else{
252  *timerNext_ms = 0;
253  }
254  }
255 
256 
257  /* CAN passive flag */
258  CANpassive = 0;
259  if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_PASSIVE) || CO_isError(NMT->emPr->em, CO_EM_CAN_RX_BUS_PASSIVE))
260  CANpassive = 1;
261 
262 
263  /* CANopen green RUN LED (DR 303-3) */
264  switch(NMT->operatingState){
265  case CO_NMT_STOPPED: NMT->LEDgreenRun = NMT->LEDsingleFlash; break;
266  case CO_NMT_PRE_OPERATIONAL: NMT->LEDgreenRun = NMT->LEDblinking; break;
267  case CO_NMT_OPERATIONAL: NMT->LEDgreenRun = 1; break;
268  }
269 
270 
271  /* CANopen red ERROR LED (DR 303-3) */
272  if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_OFF))
273  NMT->LEDredError = 1;
274 
275  else if(CO_isError(NMT->emPr->em, CO_EM_SYNC_TIME_OUT))
276  NMT->LEDredError = NMT->LEDtripleFlash;
277 
278  else if(CO_isError(NMT->emPr->em, CO_EM_HEARTBEAT_CONSUMER) || CO_isError(NMT->emPr->em, CO_EM_HB_CONSUMER_REMOTE_RESET))
279  NMT->LEDredError = NMT->LEDdoubleFlash;
280 
281  else if(CANpassive || CO_isError(NMT->emPr->em, CO_EM_CAN_BUS_WARNING))
282  NMT->LEDredError = NMT->LEDsingleFlash;
283 
284  else if(errorRegister)
285  NMT->LEDredError = (NMT->LEDblinking>=0)?-1:1;
286 
287  else
288  NMT->LEDredError = -1;
289 
290 
291  /* in case of error enter pre-operational state */
292  if(errorBehavior && (NMT->operatingState == CO_NMT_OPERATIONAL)){
293  if(CANpassive && (errorBehavior[2] == 0 || errorBehavior[2] == 2)) errorRegister |= 0x10;
294 
295  if(errorRegister){
296  /* Communication error */
297  if(errorRegister & CO_ERR_REG_COMM_ERR){
298  if(errorBehavior[1] == 0){
299  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
300  }
301  else if(errorBehavior[1] == 2){
302  NMT->operatingState = CO_NMT_STOPPED;
303  }
304  else if(CO_isError(NMT->emPr->em, CO_EM_CAN_TX_BUS_OFF)
305  || CO_isError(NMT->emPr->em, CO_EM_HEARTBEAT_CONSUMER)
306  || CO_isError(NMT->emPr->em, CO_EM_HB_CONSUMER_REMOTE_RESET))
307  {
308  if(errorBehavior[0] == 0){
309  NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
310  }
311  else if(errorBehavior[0] == 2){
312  NMT->operatingState = CO_NMT_STOPPED;
313  }
314  }
315  }
316 
317  /* Generic error */
318  if(errorRegister & CO_ERR_REG_GENERIC_ERR){
319  if (errorBehavior[3] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
320  else if (errorBehavior[3] == 2) NMT->operatingState = CO_NMT_STOPPED;
321  }
322 
323  /* Device profile error */
324  if(errorRegister & CO_ERR_REG_DEV_PROFILE){
325  if (errorBehavior[4] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
326  else if (errorBehavior[4] == 2) NMT->operatingState = CO_NMT_STOPPED;
327  }
328 
329  /* Manufacturer specific error */
330  if(errorRegister & CO_ERR_REG_MANUFACTURER){
331  if (errorBehavior[5] == 0) NMT->operatingState = CO_NMT_PRE_OPERATIONAL;
332  else if (errorBehavior[5] == 2) NMT->operatingState = CO_NMT_STOPPED;
333  }
334 
335  /* if operational state is lost, send HB immediately. */
336  if(NMT->operatingState != CO_NMT_OPERATIONAL)
337  NMT->HBproducerTimer = HBtime;
338  }
339  }
340 
341  if(NMT->pFunctNMT!=NULL && currentOperatingState!=NMT->operatingState){
342  NMT->pFunctNMT(NMT->operatingState);
343  }
344 
345  return NMT->resetCommand;
346 }
